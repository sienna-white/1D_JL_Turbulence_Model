744 function nc_put_var(ncid::Integer,varid::Integer,data)
745 dimids = nc_inq_vardimid(ncid,varid)
746 ndims = length(dimids)
747 ncsize = ntuple(i -> nc_inq_dimlen(ncid,dimids[ndims-i+1]), ndims)
748
749 if isempty(nc_inq_unlimdims(ncid))
750 if ncsize != size(data)
751 path = nc_inq_path(ncid)
752 varname = nc_inq_varname(ncid,varid)
753 throw(NetCDFError(-1,"wrong size of variable '$varname' (size $ncsize) in file '$path' for an array of size $(size(data))"))
754 end
755
756 nc_unsafe_put_var(ncid,varid,data)
757 else
758 # honor this good advice:
759
760 # Take care when using this function with record variables (variables
761 # that use the ::NC_UNLIMITED dimension). If you try to write all the
762 # values of a record variable into a netCDF file that has no record data
763 # yet (hence has 0 records), nothing will be written. Similarly, if you
764 # try to write all the values of a record variable but there are more
765 # records in the file than you assume, more in-memory data will be
766 # accessed than you supply, which may result in a segmentation
767 # violation. To avoid such problems, it is better to use the nc_put_vara
768 # interfaces for variables that use the ::NC_UNLIMITED dimension.
769
770 # https://github.com/Unidata/netcdf-c/blob/48cc56ea3833df455337c37186fa6cd7fac9dc7e/libdispatch/dvarput.c#L895
771
772 startp = zeros(ndims)
773 countp = Int[reverse(size(data))...,]
774 nc_put_vara(ncid,varid,startp,countp,data)
775 end
776 end

931 function _nc_check_size_put_vars(ncid,varid,countp,op)
932 # dimension index for NetCDF variable
933 i1 = 1
934 # dimension index of data op
935 i2 = 1
936
937 # not_one(x) = x != 1
938 # if filter(not_one,reverse(countp)) != filter(not_one,[size(op)...])
939 # path = nc_inq_path(ncid)
940 # varname = nc_inq_varname(ncid,varid)
941 # throw(DimensionMismatch("size mismatch for variable '$(varname)' in file '$(path)'. Trying to write $(size(op)) elements while $(countp) are expected"))
942 # end
943
944 unlimdims = nc_inq_unlimdims(ncid)
945 dimids = nc_inq_vardimid(ncid,varid)
946
947 countp = reverse(countp)
948 dimids = reverse(dimids)
949
950 while true
951 if (i2 > ndims(op)) && (i1 > length(countp))
952 break
953 end
954
955 count_i1 =
956 if i1 <= length(countp)
957 countp[i1]
958 else
959 1
960 end
961
962 # ignore dimensions with only one element
963 if (count_i1 == 1) && (i1 <= length(countp))
964 i1 += 1
965 continue
966 end
967 if size(op,i2) == 1 && (i2 <= ndims(op))
968 i2 += 1
969 continue
970 end
971
972 # no test for unlimited dimensions
973 if (i1 <= length(dimids)) && (dimids[i1] in unlimdims)
974 # ok
975 elseif (size(op,i2) !== count_i1)
976 path = nc_inq_path(ncid)
977 varname = nc_inq_varname(ncid,varid)
978
979 throw(NetCDFError(NC_EEDGE,"size mismatch for variable '$(varname)' in file '$(path)'. Trying to write $(size(op)) elements while $(countp) are expected"))
980 end
981
982 i1 += 1
983 i2 += 1
984 end
985 end
