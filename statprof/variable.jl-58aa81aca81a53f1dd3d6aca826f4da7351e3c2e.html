161 """
162 data = loadragged(ncvar,index::Union{Colon,UnitRange,Integer})
163
164 Load data from `ncvar` in the [contiguous ragged array representation](https://web.archive.org/web/20190111092546/http://cfconventions.org/cf-conventions/v1.6.0/cf-conventions.html#_contiguous_ragged_array_representation) as a
165 vector of vectors. It is typically used to load a list of profiles
166 or time series of different length each.
167
168 The [indexed ragged array representation](https://web.archive.org/web/20190111092546/http://cfconventions.org/cf-conventions/v1.6.0/cf-conventions.html#_indexed_ragged_array_representation) is currently not supported.
169 """
170 function loadragged(ncvar,index::Union{Colon,UnitRange})
171 ds = dataset(ncvar)
172
173 dimensionnames = dimnames(ncvar)
174 if length(dimensionnames) !== 1
175 throw(NetCDFError(-1, "NetCDF variable $(name(ncvar)) should have only one dimensions"))
176 end
177 dimname = dimensionnames[1]
178
179 ncvarsizes = varbyattrib(ds,sample_dimension = dimname)
180 if length(ncvarsizes) !== 1
181 throw(NetCDFError(-1, "There should be exactly one NetCDF variable with the attribute 'sample_dimension' equal to '$(dimname)'"))
182 end
183
184 # ignore _FillValue which can be 0 for WOD
185 ncvarsize = ncvarsizes[1].var
186
187 isa(index,Colon)||(index[1]==1) ? n0=1 : n0=1+sum(ncvarsize[1:index[1]-1])
188 isa(index,Colon) ? n1=sum(ncvarsize[:]) : n1=sum(ncvarsize[1:index[end]])
189
190 varsize = ncvarsize[index]
191
192 istart = 0;
193 tmp = ncvar[n0:n1]
194
195 T = typeof(view(tmp,1:varsize[1]))
196 data = Vector{T}(undef,length(varsize))
197
198 for i in eachindex(varsize,data)
199 data[i] = view(tmp,istart+1:istart+varsize[i]);
200 istart += varsize[i]
201 end
202 return data
203 end

499 function to_range_list(index::Vector{T},len) where T <: Integer
500 grow(istart) = istart[begin]:(istart[end]+step(istart))
501
502 baseindex = 1
503 indices_ranges = UnitRange{T}[]
504
505 while baseindex <= length(index)
506 range = index[baseindex]:index[baseindex]
507 range_test = grow(range)
508 index_view = @view index[baseindex:end]
509
510 while checkbounds(Bool,index_view,length(range_test)) &&
511 (range_test[end] == index_view[length(range_test)])
512
513 range = range_test
514 range_test = grow(range_test)
515 end
516
517 push!(indices_ranges,range)
518 baseindex += length(range)
519 end
520
521 @assert reduce(vcat,indices_ranges,init=T[]) == index
522 return indices_ranges
523 end
