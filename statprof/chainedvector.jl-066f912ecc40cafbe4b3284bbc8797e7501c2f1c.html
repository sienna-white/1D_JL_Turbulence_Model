667 i isa Bool && throw(ArgumentError("invalid index: $i of type Bool"))

683 function Base.vcat(A::ChainedVector{T, AT}, arrays::ChainedVector{T, AT}...) where {T, AT <: AbstractVector{T}}
684 newarrays = vcat(A.arrays, map(x->x.arrays, arrays)...)
685 n = length(A.inds)
686 inds = Vector{Int}(undef, n + sum(x->length(x.inds), arrays))
687 copyto!(inds, 1, A.inds, 1, n)
688 m = n + 1
689 for x in arrays
690 for y in x.arrays
691 @inbounds inds[m] = ((m - 1) == 0 ? 0 : inds[m - 1]) + length(y)
692 m += 1
693 end
694 end

788 function Base.map!(f::F, x::ChainedVector, y::ChainedVector{T}) where {F, T}
789 length(x) >= length(y) || throw(ArgumentError("destination must be at least as long as map! source"))
790 # check for potential fastpath
791 N = length(y.arrays)
792 if length(x.arrays) == N
793 match = true
794 for i = 1:N
795 @inbounds match &= length(x.arrays[i]) == length(y.arrays[i])
796 end
797 if match
798 # common when x was created like `similar(y, length(x))`
799 for i = 1:length(x.arrays)
800 @inbounds map!(f, x.arrays[i], y.arrays[i])
801 end
802 return x
803 end
804 end
805 # slower path
806 cleanup!(y)
807 yidx = yi = 1
808 @inbounds ychunk = y.arrays[yidx]
809 ychunklen = length(ychunk)
810 for array in x.arrays
811 for j in eachindex(array)
812 @inbounds array[j] = f(ychunk[yi])
813 yi += 1
814 if yi > ychunklen
815 yi = 1
816 yidx += 1
817 yidx > N && @goto done
818 @inbounds ychunk = y.arrays[yidx]
819 ychunklen = length(ychunk)
820 end
821 end
822 end
823 @label done
824 return x
825 end

858 function Base.findmax(f::F, x::ChainedVector) where {F}
859 isempty(x) && throw(ArgumentError("collection must be non-empty"))
860 cleanup!(x) # get rid of any empty arrays
861 i = 1
862 y = f(x.arrays[1][1])
863 # x > y iff y < x for a well ordered set
864 # nb. isgreater = !isless is not correct. That is `>=`
865 isgreater(x, y) = isless(y, x)

877 function findXwithfirst(comp, f, x, y, i)
878 i′ = 1
879 for A in x.arrays
880 for y′ in A
881 y′′ = f(y′)
882 c = comp(y′′, y) # store this before y changes
883 y = ifelse(c, y′′, y)
884 i = ifelse(c, i′, i)
885 i′ += 1
886 end
887 end
888 return y, i
889 end
