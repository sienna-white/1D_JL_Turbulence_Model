713 replace(new::Union{Function, Type}, A; [count::Integer])
714
715 Return a copy of `A` where each value `x` in `A` is replaced by `new(x)`.
716 If `count` is specified, then replace at most `count` values in total
717 (replacements being defined as `new(x) !== x`).
718
719 !!! compat "Julia 1.7"
720 Version 1.7 is required to replace elements of a `Tuple`.
721
722 # Examples
723 ```jldoctest
724 julia> replace(x -> isodd(x) ? 2x : x, [1, 2, 3, 4])
725 4-element Vector{Int64}:
726 2
727 2
728 6
729 4
730
731 julia> replace(Dict(1=>2, 3=>4)) do kv
732 first(kv) < 3 ? first(kv)=>3 : kv
733 end
734 Dict{Int64, Int64} with 2 entries:
735 3 => 4
736 1 => 3
737 ```
738 """
739 replace(new::Callable, A; count::Integer=typemax(Int)) =
740 _replace!(new, _similar_or_copy(A), A, check_count(count))
741
742 # Handle ambiguities
743 replace!(a::Callable, b::Pair; count::Integer=-1) = throw(MethodError(replace!, (a, b)))
744 replace!(a::Callable, b::Pair, c::Pair; count::Integer=-1) = throw(MethodError(replace!, (a, b, c)))
745 replace(a::Callable, b::Pair; count::Integer=-1) = throw(MethodError(replace, (a, b)))
746 replace(a::Callable, b::Pair, c::Pair; count::Integer=-1) = throw(MethodError(replace, (a, b, c)))

748 ### replace! for AbstractDict/AbstractSet
749
750 askey(k, ::AbstractDict) = k.first
751 askey(k, ::AbstractSet) = k
752
753 function _replace!(new::Callable, res::Union{AbstractDict,AbstractSet},
754 A::Union{AbstractDict,AbstractSet}, count::Int)
755 @assert res isa AbstractDict && A isa AbstractDict ||
756 res isa AbstractSet && A isa AbstractSet
757 count == 0 && return res
758 c = 0
759 if res === A # cannot replace elements while iterating over A
760 repl = Pair{eltype(A),eltype(A)}[]
761 for x in A
762 y = new(x)
763 if x !== y
764 push!(repl, x => y)
765 c += 1
766 c == count && break
767 end
768 end
769 for oldnew in repl
770 pop!(res, askey(first(oldnew), res))
771 end
772 for oldnew in repl
773 push!(res, last(oldnew))
774 end
775 else
776 for x in A
777 y = new(x)
778 if x !== y
779 pop!(res, askey(x, res))
780 push!(res, y)
781 c += 1
782 c == count && break
783 end
784 end
785 end
786 res
787 end

791 function _replace!(new::Callable, res::AbstractArray, A::AbstractArray, count::Int)
792 c = 0
793 if count >= length(A) # simpler loop allows for SIMD
794 if res === A # for optimization only
795 for i in eachindex(A)
796 @inbounds Ai = A[i]
797 y = new(Ai)
798 @inbounds A[i] = y
799 end
800 else
801 for i in eachindex(A)
802 @inbounds Ai = A[i]
803 y = new(Ai)
804 @inbounds res[i] = y
805 end
806 end
807 else
808 for i in eachindex(A)
809 @inbounds Ai = A[i]
810 if c < count
811 y = new(Ai)
812 @inbounds res[i] = y
813 c += (Ai !== y)
814 else
815 @inbounds res[i] = Ai
816 end
817 end
818 end
819 res
820 end

822 ### specialization for Dict / Set
823
824 function _replace!(new::Callable, t::Dict{K,V}, A::AbstractDict, count::Int) where {K,V}
825 # we ignore A, which is supposed to be equal to the destination t,
826 # as it can generally be faster to just replace inline
827 count == 0 && return t
828 c = 0
829 news = Pair{K,V}[]
830 i = skip_deleted_floor!(t)
831 @inbounds while i != 0
832 k1, v1 = t.keys[i], t.vals[i]
833 x1 = Pair{K,V}(k1, v1)
834 x2 = new(x1)
835 if x1 !== x2
836 k2, v2 = first(x2), last(x2)
837 if isequal(k1, k2)
838 t.keys[i] = k2
839 t.vals[i] = v2
840 t.age += 1
841 else
842 _delete!(t, i)
843 push!(news, x2)
844 end
845 c += 1
846 c == count && break
847 end
848 i = i == typemax(Int) ? 0 : skip_deleted(t, i+1)
849 end
850 for n in news
851 push!(t, n)
852 end
853 t
854 end

856 function _replace!(new::Callable, t::Set{T}, ::AbstractSet, count::Int) where {T}
857 _replace!(t.dict, t.dict, count) do kv
858 k = first(kv)
859 k2 = new(k)
860 k2 === k ? kv : k2 => nothing
861 end
862 t
863 end

867 function _replace(f::Callable, t::Tuple, count::Int)
868 if count == 0 || isempty(t)
869 t
870 else
871 x = f(t[1])
872 (x, _replace(f, tail(t), count - !==(x, t[1]))...)
873 end
874 end

876 replace(f::Callable, t::Tuple; count::Integer=typemax(Int)) =
877 _replace(f, t, check_count(count))
878
879 function _replace(t::Tuple, count::Int, old_new::Tuple{Vararg{Pair}})
880 _replace(t, count) do x
881 @inline
882 for o_n in old_new
883 isequal(first(o_n), x) && return last(o_n)
884 end
885 return x
886 end
887 end

889 replace(t::Tuple, old_new::Pair...; count::Integer=typemax(Int)) =
890 _replace(t, check_count(count), old_new)
