451 function readbytes_all!(s::IOStream,
452 b::Union{Array{UInt8}, FastContiguousSubArray{UInt8,<:Any,<:Array{UInt8}}},
453 nb::Integer)
454 olb = lb = length(b)
455 nr = 0
456 let l = s._dolock, slock = s.lock
457 l && lock(slock)
458 GC.@preserve b while nr < nb
459 if lb < nr+1
460 try
461 lb = max(65536, (nr+1) * 2)
462 resize!(b, lb)
463 catch
464 l && unlock(slock)
465 rethrow()
466 end
467 end
468 thisr = Int(ccall(:ios_readall, Csize_t, (Ptr{Cvoid}, Ptr{Cvoid}, Csize_t),
469 s.ios, pointer(b, nr+1), min(lb-nr, nb-nr)))
470 nr += thisr
471 (nr == nb || thisr == 0 || _eof_nolock(s)) && break
472 end
473 l && unlock(slock)
474 end
475 if lb > olb && lb > nr
476 resize!(b, max(olb, nr)) # shrink to just contain input data if was resized
477 end

502
503 Read at most `nb` bytes from `stream` into `b`, returning the number of bytes read.
504 The size of `b` will be increased if needed (i.e. if `nb` is greater than `length(b)`
505 and enough bytes could be read), but it will never be decreased.
506
507 If `all` is `true` (the default), this function will block repeatedly trying to read all
508 requested bytes, until an error or end-of-file occurs. If `all` is `false`, at most one
509 `read` call is performed, and the amount of data returned is device-dependent. Note that not
510 all stream types support the `all` option.
511 """

514 nb=length(b);
515 all::Bool=true)
516 return all ? readbytes_all!(s, b, nb) : readbytes_some!(s, b, nb)

519 function read(s::IOStream)
520 # First we try to fill the buffer. If that gives us the whole file,
521 # copy it out and return. Otherwise look at the file size and use it
522 # to prealloate space. Determining the size requires extra syscalls,
523 # which we want to avoid for small files.
524 @_lock_ios s begin
525 nb = ccall(:ios_fillbuf, Cssize_t, (Ptr{Cvoid},), s.ios)
526 if nb != -1
527 b = StringVector(nb)
528 readbytes_all!(s, b, nb)
529 else
530 sz = try # filesize is just a hint, so ignore if it fails
531 filesize(s)
532 catch ex
533 ex isa IOError || rethrow()
534 Int64(-1)
535 end
536 if sz > 0
537 pos = position(s)
538 if pos > 0
539 sz -= pos
540 end
541 end
542 b = StringVector(sz < 0 ? 1024 : sz)
543 nr = readbytes_all!(s, b, sz < 0 ? typemax(Int) : sz)
544 resize!(b, nr)
545 end
546 end
547 return b
548 end

554
555 If `all` is `true` (the default), this function will block repeatedly trying to read all
