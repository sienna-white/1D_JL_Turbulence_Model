165 @constprop :none function rehash!(h::Dict{K,V}, newsz = length(h.keys)) where V where K

167 oldk = h.keys
168 oldv = h.vals
169 sz = length(olds)
170 newsz = _tablesz(newsz)
171 h.age += 1
172 h.idxfloor = 1
173 if h.count == 0
174 resize!(h.slots, newsz)
175 fill!(h.slots, 0x0)
176 resize!(h.keys, newsz)
177 resize!(h.vals, newsz)
178 h.ndel = 0

183 slots = zeros(UInt8,newsz)
184 keys = Vector{K}(undef, newsz)
185 vals = Vector{V}(undef, newsz)
186 age0 = h.age
187 count = 0
188 maxprobe = 0
189
190 for i = 1:sz
191 @inbounds if (olds[i] & 0x80) != 0
192 k = oldk[i]
193 v = oldv[i]
194 index, sh = hashindex(k, newsz)
195 index0 = index

219 function sizehint!(d::Dict{T}, newsz) where T
220 oldsz = length(d.slots)
221 # limit new element count to max_values of the key type
222 newsz = min(max(newsz, length(d)), max_values(T)::Int)
223 # need at least 1.5n space to hold n elements
224 newsz = _tablesz(cld(3 * newsz, 2))
225 return newsz == oldsz ? d : rehash!(d, newsz)
226 end

753 function map!(f, iter::ValueIterator{<:Dict})
754 dict = iter.dict
755 vals = dict.vals
756 # @inbounds is here so that it gets propagated to isslotfilled
757 @inbounds for i = dict.idxfloor:lastindex(vals)
758 if isslotfilled(dict, i)
759 vals[i] = f(vals[i])
760 end
761 end
762 return iter

765 function mergewith!(combine, d1::Dict{K, V}, d2::AbstractDict) where {K, V}
766 haslength(d2) && sizehint!(d1, length(d1) + length(d2))
767 for (k, v) in d2
768 i, sh = ht_keyindex2_shorthash!(d1, k)
769 if i > 0
770 d1.vals[i] = combine(d1.vals[i], v)
771 else
772 if !(k isa K)
773 k1 = convert(K, k)::K
774 if !isequal(k, k1)
775 throw(ArgumentError("$(limitrepr(k)) is not a valid key for type $K"))
776 end
777 k = k1
778 end
779 if !isa(v, V)
780 v = convert(V, v)::V
781 end
782 @inbounds _setindex!(d1, v, k, -i, sh)
783 end
784 end
785 return d1
786 end

788 struct ImmutableDict{K,V} <: AbstractDict{K,V}
789 parent::ImmutableDict{K,V}
790 key::K
791 value::V
792 ImmutableDict{K,V}() where {K,V} = new() # represents an empty dictionary
793 ImmutableDict{K,V}(key, value) where {K,V} = (empty = new(); new(empty, key, value))
794 ImmutableDict{K,V}(parent::ImmutableDict, key, value) where {K,V} = new(parent, key, value)
795 end
796
797 """
798 ImmutableDict
799
800 `ImmutableDict` is a dictionary implemented as an immutable linked list,
801 which is optimal for small dictionaries that are constructed over many individual insertions.
802 Note that it is not possible to remove a value, although it can be partially overridden and hidden
803 by inserting a new value with the same key.
804
805 ImmutableDict(KV::Pair)
806
807 Create a new entry in the `ImmutableDict` for a `key => value` pair
808
809 - use `(key => value) in dict` to see if this particular combination is in the properties set
810 - use `get(dict, key, default)` to retrieve the most recent value for a particular key
811
812 """
813 ImmutableDict
814 ImmutableDict(KV::Pair{K,V}) where {K,V} = ImmutableDict{K,V}(KV[1], KV[2])
815 ImmutableDict(t::ImmutableDict{K,V}, KV::Pair) where {K,V} = ImmutableDict{K,V}(t, KV[1], KV[2])
816 ImmutableDict(t::ImmutableDict{K,V}, KV::Pair, rest::Pair...) where {K,V} =
817 ImmutableDict(ImmutableDict(t, KV), rest...)
818 ImmutableDict(KV::Pair, rest::Pair...) = ImmutableDict(ImmutableDict(KV), rest...)

820 function in(key_value::Pair, dict::ImmutableDict, valcmp=(==))
821 key, value = key_value
822 while isdefined(dict, :parent)
823 if isequal(dict.key, key)
824 valcmp(value, dict.value) && return true
825 end
826 dict = dict.parent
827 end
828 return false
829 end
