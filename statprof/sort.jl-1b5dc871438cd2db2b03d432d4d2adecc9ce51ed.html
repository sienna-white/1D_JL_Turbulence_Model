937 """
938 RadixSort <: Algorithm
939
940 Use the radix sort algorithm.
941
942 `RadixSort` is a stable least significant bit first radix sort algorithm that runs in
943 `O(length * log(range))` time and linear space.
944
945 It first sorts the entire vector by the last `chunk_size` bits, then by the second
946 to last `chunk_size` bits, and so on. Stability means that it will not reorder two elements
947 that compare equal. This is essential so that the order introduced by earlier,
948 less significant passes is preserved by later passes.
949
950 Each pass divides the input into `2^chunk_size == mask+1` buckets. To do this, it
951 * counts the number of entries that fall into each bucket
952 * uses those counts to compute the indices to move elements of those buckets into
953 * moves elements into the computed indices in the swap array
954 * switches the swap and working array
955
956 `chunk_size` is larger for larger inputs and determined by an empirical heuristic.
957 """
958 struct RadixSort <: Algorithm end
959 function _sort!(v::AbstractVector, a::RadixSort, o::DirectOrdering, kw)
960 @getkw lo hi mn mx scratch
961 umn = uint_map(mn, o)
962 urange = uint_map(mx, o)-umn
963 bits = unsigned(top_set_bit(urange))
964
965 # At this point, we are committed to radix sort.
966 u = uint_map!(v, lo, hi, o)
967
968 # we subtract umn to avoid radixing over unnecessary bits. For example,
969 # Int32[3, -1, 2] uint_maps to UInt32[0x80000003, 0x7fffffff, 0x80000002]
970 # which uses all 32 bits, but once we subtract umn = 0x7fffffff, we are left with
971 # UInt32[0x00000004, 0x00000000, 0x00000003] which uses only 3 bits, and
972 # Float32[2.012, 400.0, 12.345] uint_maps to UInt32[0x3fff3b63, 0x3c37ffff, 0x414570a4]
973 # which is reduced to UInt32[0x03c73b64, 0x00000000, 0x050d70a5] using only 26 bits.
974 # the overhead for this subtraction is small enough that it is worthwhile in many cases.
975
976 # this is faster than u[lo:hi] .-= umn as of v1.9.0-DEV.100
977 @inbounds for i in lo:hi
978 u[i] -= umn
979 end
980
981 scratch, t = make_scratch(scratch, eltype(v), hi-lo+1)
982 tu = reinterpret(eltype(u), t)
983 if radix_sort!(u, lo, hi, bits, tu, 1-lo)
984 uint_unmap!(v, u, lo, hi, o, umn)
985 else
986 uint_unmap!(v, tu, lo, hi, o, umn, 1-lo)
987 end
988 scratch
989 end

1027 # to the pivot, and store the result in t[lo:hi].

1029 # sets `pivot_dest[pivot_index+pivot_index_offset] = pivot` and returns that index.
1030 function partition!(t::AbstractVector, lo::Integer, hi::Integer, offset::Integer, o::Ordering,
1031 v::AbstractVector, rev::Bool, pivot_dest::AbstractVector, pivot_index_offset::Integer)
1032 # Ideally we would use `pivot_index = rand(lo:hi)`, but that requires Random.jl
1033 # and would mutate the global RNG in sorting.
1034 pivot_index = typeof(hi-lo)(hash(lo) % (hi-lo+1)) + lo
1035 @inbounds begin
1036 pivot = v[pivot_index]
1037 while lo < pivot_index
1038 x = v[lo]
1039 fx = rev ? !lt(o, x, pivot) : lt(o, pivot, x)
1040 t[(fx ? hi : lo) - offset] = x
1041 offset += fx
1042 lo += 1
1043 end
1044 while lo < hi
1045 x = v[lo+1]
1046 fx = rev ? lt(o, pivot, x) : !lt(o, x, pivot)
1047 t[(fx ? hi : lo) - offset] = x
1048 offset += fx
1049 lo += 1
1050 end
1051 pivot_index = lo-offset + pivot_index_offset
1052 pivot_dest[pivot_index] = pivot
1053 end
1054
1055 # t_pivot_index = lo-offset (i.e. without pivot_index_offset)
1056 # t[t_pivot_index] is whatever it was before unless t is the pivot_dest
1057 # t[<t_pivot_index] <* pivot, stable
1058 # t[>t_pivot_index] >* pivot, reverse stable

1065 @getkw lo hi scratch
1066
1067 if t === nothing
1068 scratch, t = make_scratch(scratch, eltype(v), hi-lo+1)
1069 offset = 1-lo
1070 kw = (;kw..., scratch)
1071 end

1073 while lo < hi && hi - lo > SMALL_THRESHOLD
1074 j = if swap
1075 partition!(v, lo+offset, hi+offset, offset, o, t, rev, v, 0)
1076 else

1078 end

1080

1135 # The return value indicates whether v is sorted (true) or t is sorted (false)
1136 # This is one of the many reasons radix_sort! is not exported.
1137 function radix_sort!(v::AbstractVector{U}, lo::Integer, hi::Integer, bits::Unsigned,
1138 t::AbstractVector{U}, offset::Integer,
1139 chunk_size=radix_chunk_size_heuristic(lo, hi, bits)) where U <: Unsigned
1140 # bits is unsigned for performance reasons.
1141 counts = Vector{Int}(undef, 1 << chunk_size + 1) # TODO use scratch for this
1142
1143 shift = 0
1144 while true
1145 @noinline radix_sort_pass!(t, lo, hi, offset, counts, v, shift, chunk_size)
1146 # the latest data resides in t
1147 shift += chunk_size
1148 shift < bits || return false
1149 @noinline radix_sort_pass!(v, lo+offset, hi+offset, -offset, counts, t, shift, chunk_size)
1150 # the latest data resides in v
1151 shift += chunk_size
1152 shift < bits || return true
1153 end
1154 end
1155 function radix_sort_pass!(t, lo, hi, offset, counts, v, shift, chunk_size)
1156 mask = UInt(1) << chunk_size - 1 # mask is defined in pass so that the compiler
1157 @inbounds begin # ↳ knows it's shape
1158 # counts[2:mask+2] will store the number of elements that fall into each bucket.
1159 # if chunk_size = 8, counts[2] is bucket 0x00 and counts[257] is bucket 0xff.
1160 counts .= 0
1161 for k in lo:hi
1162 x = v[k] # lookup the element
1163 i = (x >> shift)&mask + 2 # compute its bucket's index for this pass
1164 counts[i] += 1 # increment that bucket's count
1165 end
1166
1167 counts[1] = lo + offset # set target index for the first bucket
1168 cumsum!(counts, counts) # set target indices for subsequent buckets
1169 # counts[1:mask+1] now stores indices where the first member of each bucket
1170 # belongs, not the number of elements in each bucket. We will put the first element
1171 # of bucket 0x00 in t[counts[1]], the next element of bucket 0x00 in t[counts[1]+1],
1172 # and the last element of bucket 0x00 in t[counts[2]-1].
1173
1174 for k in lo:hi
1175 x = v[k] # lookup the element
1176 i = (x >> shift)&mask + 1 # compute its bucket's index for this pass
1177 j = counts[i] # lookup the target index
1178 t[j] = x # put the element where it belongs
1179 counts[i] = j + 1 # increment the target index for the next
1180 end # ↳ element in this bucket
1181 end
1182 end
1183 function radix_chunk_size_heuristic(lo::Integer, hi::Integer, bits::Unsigned)
1184 # chunk_size is the number of bits to radix over at once.
1185 # We need to allocate an array of size 2^chunk size, and on the other hand the higher
1186 # the chunk size the fewer passes we need. Theoretically, chunk size should be based on
1187 # the Lambert W function applied to length. Empirically, we use this heuristic:
1188 guess = min(10, log(maybe_unsigned(hi-lo))*3/4+3)
1189 # TODO the maximum chunk size should be based on architecture cache size.
1190
1191 # We need iterations * chunk size ≥ bits, and these cld's
1192 # make an effort to get iterations * chunk size ≈ bits
1193 UInt8(cld(bits, cld(bits, guess)))
1194 end

1200 @inbounds for i in (lo+1):hi
1201 lt(o, v[i], v[i-1]) && return false

1247 triggered and can enable other optimizations to be applied later. For example,
1248 BoolOptimization cannot apply to an `AbstractVector{Union{Missing, Bool}}`, but after
1249 [`MissingOptimization`](@ref) is applied, that input will be converted into am
1250 `AbstractVector{Bool}`.
1251
1252 We next apply [`BoolOptimization`](@ref) because it also has no runtime cost when it is not
1253 triggered and when it is triggered, it is an incredibly efficient algorithm (sorting `Bool`s
1254 is quite easy).
1255
1256 Next, we dispatch to [`InsertionSort`](@ref) for inputs with `length <= 10`. This dispatch
1257 occurs before the [`IEEEFloatOptimization`](@ref) pass because the
1258 [`IEEEFloatOptimization`](@ref)s are not beneficial for very small inputs.
1259
1260 To conclude the [`InitialOptimizations`](@ref), we apply [`IEEEFloatOptimization`](@ref).
1261
1262 After these optimizations, we branch on whether radix sort and related algorithms can be
1263 applied to the input vector and ordering. We conduct this branch by testing if
1264 `UIntMappable(v, order) !== nothing`. That is, we see if we know of a reversible mapping
1265 from `eltype(v)` to `UInt` that preserves the ordering `order`. We perform this check after
1266 the initial optimizations because they can change the input vector's type and ordering to
1267 make them `UIntMappable`.
1268
1269 If the input is not [`UIntMappable`](@ref), then we perform a presorted check and dispatch
1270 to [`ScratchQuickSort`](@ref).
1271
1272 Otherwise, we dispatch to [`InsertionSort`](@ref) for inputs with `length <= 40` and then
1273 perform a presorted check ([`CheckSorted`](@ref)).
1274
1275 We check for short inputs before performing the presorted check to avoid the overhead of the
1276 check for small inputs. Because the alternate dispatch is to [`InsertionSort`](@ref) which
1277 has efficient `O(n)` runtime on presorted inputs, the check is not necessary for small
1278 inputs.
1279
1280 We check if the input is reverse-sorted for long vectors (more than 500 elements) because
1281 the check is essentially free unless the input is almost entirely reverse sorted.
1282
1283 Note that once the input is determined to be [`UIntMappable`](@ref), we know the order forms
1284 a [total order](wikipedia.org/wiki/Total_order) over the inputs and so it is impossible to
1285 perform an unstable sort because no two elements can compare equal unless they _are_ equal,
1286 in which case switching them is undetectable. We utilize this fact to perform a more
1287 aggressive reverse sorted check that will reverse the vector `[3, 2, 2, 1]`.
1288
1289 After these potential fast-paths are tried and failed, we [`ComputeExtrema`](@ref) of the
1290 input. This computation has a fairly fast `O(n)` runtime, but we still try to delay it until
1291 it is necessary.
1292
1293 Next, we [`ConsiderCountingSort`](@ref). If the range the input is small compared to its
1294 length, we apply [`CountingSort`](@ref).
1295
1296 Next, we [`ConsiderRadixSort`](@ref). This is similar to the dispatch to counting sort,
1297 but we conside rthe number of _bits_ in the range, rather than the range itself.
1298 Consequently, we apply [`RadixSort`](@ref) for any reasonably long inputs that reach this
1299 stage.
1300
1301 Finally, if the input has length less than 80, we dispatch to [`InsertionSort`](@ref) and
1302 otherwise we dispatch to [`ScratchQuickSort`](@ref).
1303 """

1356
1357 Sort the vector `v` in place. A stable algorithm is used by default: the
1358 ordering of elements that compare equal is preserved. A specific algorithm can
1359 be selected via the `alg` keyword (see [Sorting Algorithms](@ref) for available
1360 algorithms).
1361
1362 Elements are first transformed with the function `by` and then compared
1363 according to either the function `lt` or the ordering `order`. Finally, the
1364 resulting order is reversed if `rev=true` (this preserves forward stability:
1365 elements that compare equal are not reversed). The current implemention applies
1366 the `by` transformation before each comparison rather than once per element.
1367
1368 Passing an `lt` other than `isless` along with an `order` other than
1369 [`Base.Order.Forward`](@ref) or [`Base.Order.Reverse`](@ref) is not permitted,
1370 otherwise all options are independent and can be used together in all possible
1371 combinations. Note that `order` can also include a "by" transformation, in
1372 which case it is applied after that defined with the `by` keyword. For more
1373 information on `order` values see the documentation on [Alternate
1374 Orderings](@ref).
1375
1376 Relations between two elements are defined as follows (with "less" and
1377 "greater" exchanged when `rev=true`):
1378
1379 * `x` is less than `y` if `lt(by(x), by(y))` (or `Base.Order.lt(order, by(x), by(y))`) yields true.
1380 * `x` is greater than `y` if `y` is less than `x`.
1381 * `x` and `y` are equivalent if neither is less than the other ("incomparable"
1382 is sometimes used as a synonym for "equivalent").
1383
1384 The result of `sort!` is sorted in the sense that every element is greater than
1385 or equivalent to the previous one.
1386
1387 The `lt` function must define a strict weak order, that is, it must be
1388
1389 * irreflexive: `lt(x, x)` always yields `false`,
1390 * asymmetric: if `lt(x, y)` yields `true` then `lt(y, x)` yields `false`,
1391 * transitive: `lt(x, y) && lt(y, z)` implies `lt(x, z)`,
1392 * transitive in equivalence: `!lt(x, y) && !lt(y, x)` and `!lt(y, z) && !lt(z,
1393 y)` together imply `!lt(x, z) && !lt(z, x)`. In words: if `x` and `y` are
1394 equivalent and `y` and `z` are equivalent then `x` and `z` must be
1395 equivalent.
1396
1397 For example `<` is a valid `lt` function for `Int` values but `≤` is not: it
1398 violates irreflexivity. For `Float64` values even `<` is invalid as it violates
1399 the fourth condition: `1.0` and `NaN` are equivalent and so are `NaN` and `2.0`
1400 but `1.0` and `2.0` are not equivalent.
1401
1402 See also [`sort`](@ref), [`sortperm`](@ref), [`sortslices`](@ref),
1403 [`partialsort!`](@ref), [`partialsortperm`](@ref), [`issorted`](@ref),

2060 #
2061 # Given 3 locations in an array (lo, mi, and hi), sort v[lo], v[mi], v[hi] and
2062 # choose the middle value as a pivot
2063 #
2064 # Upon return, the pivot is in v[lo], and v[hi] is guaranteed to be
2065 # greater than the pivot
2066
2067 @inline function selectpivot!(v::AbstractVector, lo::Integer, hi::Integer, o::Ordering)
2068 @inbounds begin
2069 mi = midpoint(lo, hi)
2070
2071 # sort v[mi] <= v[lo] <= v[hi] such that the pivot is immediately in place
2072 if lt(o, v[lo], v[mi])

2074 end

2078 v[hi], v[lo], v[mi] = v[lo], v[mi], v[hi]
2079 else
2080 v[hi], v[lo] = v[lo], v[hi]
2081 end
2082 end

2107 # v[k] >= pivot for k > j
