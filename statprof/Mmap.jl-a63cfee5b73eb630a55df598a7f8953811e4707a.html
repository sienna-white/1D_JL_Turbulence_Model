64 # Determine a stream's read/write mode, and return prot & flags appropriate for mmap
65 function settings(s::RawFD, shared::Bool)
66 flags = shared ? MAP_SHARED : MAP_PRIVATE
67 if s == INVALID_OS_HANDLE
68 flags |= MAP_ANONYMOUS
69 prot = PROT_READ | PROT_WRITE
70 else
71 mode = ccall(:fcntl, Cint, (RawFD, Cint, Cint...), s, F_GETFL)
72 systemerror("fcntl F_GETFL", mode == -1)
73 mode = mode & 3
74 prot = (mode == 0) ? PROT_READ : ((mode == 1) ? PROT_WRITE : (PROT_READ | PROT_WRITE))
75 if prot & PROT_READ == 0
76 throw(ArgumentError("mmap requires read permissions on the file (open with \"r+\" mode to override)"))
77 end
78 end
79 return prot, flags, (prot & PROT_WRITE) > 0
80 end

127 """
128 mmap(io::Union{IOStream,AbstractString,Mmap.AnonymousMmap}[, type::Type{Array{T,N}}, dims, offset]; grow::Bool=true, shared::Bool=true)
129 mmap(type::Type{Array{T,N}}, dims)
130
131 Create an `Array` whose values are linked to a file, using memory-mapping. This provides a
132 convenient way of working with data too large to fit in the computer's memory.
133
134 The type is an `Array{T,N}` with a bits-type element of `T` and dimension `N` that
135 determines how the bytes of the array are interpreted. Note that the file must be stored in
136 binary format, and no format conversions are possible (this is a limitation of operating
137 systems, not Julia).
138
139 `dims` is a tuple or single [`Integer`](@ref) specifying the size or length of the array.
140
141 The file is passed via the stream argument, either as an open [`IOStream`](@ref) or filename string.
142 When you initialize the stream, use `"r"` for a "read-only" array, and `"w+"` to create a
143 new array used to write values to disk.
144
145 If no `type` argument is specified, the default is `Vector{UInt8}`.
146
147 Optionally, you can specify an offset (in bytes) if, for example, you want to skip over a
148 header in the file. The default value for the offset is the current stream position for an
149 `IOStream`.
150
151 The `grow` keyword argument specifies whether the disk file should be grown to accommodate
152 the requested size of array (if the total file size is < requested array size). Write
153 privileges are required to grow the file.
154
155 The `shared` keyword argument specifies whether the resulting `Array` and changes made to it
156 will be visible to other processes mapping the same file.
157
158 For example, the following code
159
160 ```julia
161 # Create a file for mmapping
162 # (you could alternatively use mmap to do this step, too)
163 using Mmap
164 A = rand(1:20, 5, 30)
165 s = open("/tmp/mmap.bin", "w+")
166 # We'll write the dimensions of the array as the first two Ints in the file
167 write(s, size(A,1))
168 write(s, size(A,2))
169 # Now write the data
170 write(s, A)
171 close(s)
172
173 # Test by reading it back in
174 s = open("/tmp/mmap.bin") # default is read-only
175 m = read(s, Int)
176 n = read(s, Int)
177 A2 = mmap(s, Matrix{Int}, (m,n))
178 ```
179
180 creates a `m`-by-`n` `Matrix{Int}`, linked to the file associated with stream `s`.
181
182 A more portable file would need to encode the word size -- 32 bit or 64 bit -- and endianness
183 information in the header. In practice, consider encoding binary data using standard formats
184 like HDF5 (which can be used with memory-mapping).
185 """
186 function mmap(io::IO,
187 ::Type{Array{T,N}}=Vector{UInt8},
188 dims::NTuple{N,Integer}=(div(filesize(io)-position(io),sizeof(T)),),
189 offset::Integer=position(io); grow::Bool=true, shared::Bool=true) where {T,N}
190 # check inputs
191 isopen(io) || throw(ArgumentError("$io must be open to mmap"))
192 isbitstype(T) || throw(ArgumentError("unable to mmap $T; must satisfy isbitstype(T) == true"))
193
194 len = sizeof(T)
195 for l in dims
196 len, overflow = Base.Checked.mul_with_overflow(promote(len, l)...)
197 overflow && throw(ArgumentError("requested size prod($((sizeof(T), dims...))) too large, would overflow typeof(size(T)) == $(typeof(len))"))
198 end
199 len >= 0 || throw(ArgumentError("requested size must be 0, got $len"))
200 len == 0 && return Array{T}(undef, ntuple(x->0,Val(N)))
201 len < typemax(Int) - PAGESIZE || throw(ArgumentError("requested size must be < $(typemax(Int)-PAGESIZE), got $len"))
202
203 offset >= 0 || throw(ArgumentError("requested offset must be 0, got $offset"))
204
205 # shift `offset` to start of page boundary
206 offset_page::Int64 = div(offset, PAGESIZE) * PAGESIZE
207 # add (offset - offset_page) to `len` to get total length of memory-mapped region
208 mmaplen = (offset - offset_page) + len
209
210 file_desc = gethandle(io)
211 szfile = convert(Csize_t, len + offset)
212 requestedSizeLarger = false
213 if !(io isa Mmap.Anonymous)
214 @static if !Sys.isapple()
215 requestedSizeLarger = szfile > filesize(io)
216 end
217 end
218 # platform-specific mmapping
219 @static if Sys.isunix()
220 prot, flags, iswrite = settings(file_desc, shared)
221 if requestedSizeLarger
222 if iswrite
223 if grow
224 grow!(io, offset, len)
225 else
226 throw(ArgumentError("requested size $szfile larger than file size $(filesize(io)), but requested not to grow"))
227 end
228 else
229 throw(ArgumentError("unable to increase file size to $szfile due to read-only permissions"))
230 end
231 end
232 @static if Sys.isapple()
233 iswrite && grow && grow!(io, offset, len)
234 end
235 # mmap the file
236 ptr = ccall(:jl_mmap, Ptr{Cvoid}, (Ptr{Cvoid}, Csize_t, Cint, Cint, RawFD, Int64),
237 C_NULL, mmaplen, prot, flags, file_desc, offset_page)
238 systemerror("memory mapping failed", reinterpret(Int, ptr) == -1)
239 else
240 name, readonly, create = settings(io)
241 if requestedSizeLarger
242 if readonly
243 throw(ArgumentError("unable to increase file size to $szfile due to read-only permissions"))
244 elseif !grow
245 throw(ArgumentError("requested size $szfile larger than file size $(filesize(io)), but requested not to grow"))
246 end
247 end
248 handle = create ? ccall(:CreateFileMappingW, stdcall, Ptr{Cvoid}, (OS_HANDLE, Ptr{Cvoid}, DWORD, DWORD, DWORD, Cwstring),
249 file_desc, C_NULL, readonly ? PAGE_READONLY : PAGE_READWRITE, szfile >> 32, szfile & typemax(UInt32), name) :
250 ccall(:OpenFileMappingW, stdcall, Ptr{Cvoid}, (DWORD, Cint, Cwstring),
251 readonly ? FILE_MAP_READ : FILE_MAP_WRITE, true, name)
252 Base.windowserror(:mmap, handle == C_NULL)
253 ptr = ccall(:MapViewOfFile, stdcall, Ptr{Cvoid}, (Ptr{Cvoid}, DWORD, DWORD, DWORD, Csize_t),
254 handle, readonly ? FILE_MAP_READ : FILE_MAP_WRITE, offset_page >> 32, offset_page & typemax(UInt32), mmaplen)
255 Base.windowserror(:mmap, ptr == C_NULL)
256 end # os-test
257 # convert mmapped region to Julia Array at `ptr + (offset - offset_page)` since file was mapped at offset_page
258 A = unsafe_wrap(Array, convert(Ptr{T}, UInt(ptr) + UInt(offset - offset_page)), dims)
259 finalizer(A) do x
260
1 (0 %) samples spent in #3
1 (100 %) (ex.), 1 (100 %) (incl.) when called from calculate_brunt_vaisala line 68
@static if Sys.isunix()
261 1 (0 %) 1 (0 %) systemerror("munmap", ccall(:munmap, Cint, (Ptr{Cvoid}, Int), ptr, mmaplen) != 0)
262 else
263 status = ccall(:UnmapViewOfFile, stdcall, Cint, (Ptr{Cvoid},), ptr)!=0
264 status |= ccall(:CloseHandle, stdcall, Cint, (Ptr{Cvoid},), handle)!=0
265 Base.windowserror(:UnmapViewOfFile, status == 0)
266 end
267 end
268 return A
269 end

287 """
288 mmap(io, BitArray, [dims, offset])
289
290 Create a [`BitArray`](@ref) whose values are linked to a file, using memory-mapping; it has the same
291 purpose, works in the same way, and has the same arguments, as [`mmap`](@ref mmap), but
292 the byte representation is different.
293
294 # Examples
295 ```jldoctest
296 julia> using Mmap
297
298 julia> io = open("mmap.bin", "w+");
299
300 julia> B = mmap(io, BitArray, (25,30000));
301
302 julia> B[3, 4000] = true;
303
304 julia> Mmap.sync!(B);
305
306 julia> close(io);
307
308 julia> io = open("mmap.bin", "r+");
309
310 julia> C = mmap(io, BitArray, (25,30000));
311
312 julia> C[3, 4000]
313 true
314
315 julia> C[2, 4000]
316 false
317
318 julia> close(io)
319
320 julia> rm("mmap.bin")
321 ```
322 This creates a 25-by-30000 `BitArray`, linked to the file associated with stream `io`.
323 """
324 function mmap(io::IOStream, ::Type{<:BitArray}, dims::NTuple{N,Integer},
325 offset::Int64=position(io); grow::Bool=true, shared::Bool=true) where N
326 n = prod(dims)
327 nc = Base.num_bit_chunks(n)
328 chunks = mmap(io, Vector{UInt64}, (nc,), offset; grow=grow, shared=shared)
329 if !isreadonly(io)
330 chunks[end] &= Base._msk_end(n)
331 else
332 if chunks[end] != chunks[end] & Base._msk_end(n)
333 throw(ArgumentError("the given file does not contain a valid BitArray of size $(join(dims, 'x')) (open with \"r+\" mode to override)"))
334 end
335 end
336 B = BitArray{N}(undef, ntuple(i->0,Val(N))...)
337 B.chunks = chunks
338 B.len = n
339 if N != 1
340 B.dims = dims
341 end
342 return B
343 end
