111 elseif context !== nothing
112 f(IOContext(s, context), args...)

120 if x isa Float64
121 return 20
122 elseif x isa Float32
123 return 12
124 elseif x isa String || x isa SubString{String}

200 function show(
201 io :: IO,
202 mime :: MIME"text/plain",
203 str :: AbstractString;
204 limit :: Union{Int, Nothing} = nothing,
205 )
206 # compute limit in default case
207 if limit === nothing
208 get(io, :limit, false)::Bool || return show(io, str)
209 limit = max(20, displaysize(io)[2])
210 # one line in collection, seven otherwise
211 get(io, :typeinfo, nothing) === nothing && (limit *= 7)
212 end
213
214 # early out for short strings
215 len = ncodeunits(str)
216 len ≤ limit - 2 && # quote chars
217 return show(io, str)
218
219 # these don't depend on string data
220 units = codeunit(str) == UInt8 ? "bytes" : "code units"
221 skip_text(skip) = " ⋯ $skip $units ⋯ "
222 short = length(skip_text("")) + 4 # quote chars
223 chars = max(limit, short + 1) - short # at least 1 digit
224
225 # figure out how many characters to print in elided case
226 chars -= d = ndigits(len - chars) # first adjustment
227 chars += d - ndigits(len - chars) # second if needed
228 chars = max(0, chars)
229
230 # find head & tail, avoiding O(length(str)) computation
231 head = nextind(str, 0, 1 + (chars + 1) ÷ 2)
232 tail = prevind(str, len + 1, chars ÷ 2)
233
234 # threshold: min chars skipped to make elision worthwhile
235 t = short + ndigits(len - chars) - 1
236 n = tail - head # skipped code units
237 if 4t ≤ n || t ≤ n && t ≤ length(str, head, tail-1)
238 skip = skip_text(n)
239 show(io, SubString(str, 1:prevind(str, head)))
240 printstyled(io, skip; color=:light_yellow, bold=true)
241 show(io, SubString(str, tail))

313 """
314 join([io::IO,] iterator [, delim [, last]])
315
316 Join any `iterator` into a single string, inserting the given delimiter (if any) between
317 adjacent items. If `last` is given, it will be used instead of `delim` between the last
318 two items. Each item of `iterator` is converted to a string via `print(io::IOBuffer, x)`.
319 If `io` is given, the result is written to `io` rather than returned as a `String`.
320
321 # Examples
322 ```jldoctest
323 julia> join(["apples", "bananas", "pineapples"], ", ", " and ")
324 "apples, bananas and pineapples"
325
326 julia> join([1,2,3,4,5])
327 "12345"
328 ```
329 """
330 function join(io::IO, iterator, delim, last)
331 first = true
332 local prev
333 for item in iterator
334 if @isdefined prev
335 first ? (first = false) : print(io, delim)
336 print(io, prev)
337 end
338 prev = item
339 end
340 if @isdefined prev
341 first || print(io, last)
342 print(io, prev)
343 end
344 nothing
345 end

366 """
367 escape_string(str::AbstractString[, esc]; keep = ())::AbstractString
368 escape_string(io, str::AbstractString[, esc]; keep = ())::Nothing
369
370 General escaping of traditional C and Unicode escape sequences. The first form returns the
371 escaped string, the second prints the result to `io`.
372
373 Backslashes (`\\`) are escaped with a double-backslash (`"\\\\"`). Non-printable
374 characters are escaped either with their standard C escape codes, `"\\0"` for NUL (if
375 unambiguous), unicode code point (`"\\u"` prefix) or hex (`"\\x"` prefix).
376
377 The optional `esc` argument specifies any additional characters that should also be
378 escaped by a prepending backslash (`\"` is also escaped by default in the first form).
379
380 The argument `keep` specifies a collection of characters which are to be kept as
381 they are. Notice that `esc` has precedence here.
382
383 See also [`unescape_string`](@ref) for the reverse operation.
384
385 !!! compat "Julia 1.7"
386 The `keep` argument is available as of Julia 1.7.
387
388 # Examples
389 ```jldoctest
390 julia> escape_string("aaa\\nbbb")
391 "aaa\\\\nbbb"
392
393 julia> escape_string("aaa\\nbbb"; keep = '\\n')
394 "aaa\\nbbb"
395
396 julia> escape_string("\\xfe\\xff") # invalid utf-8
397 "\\\\xfe\\\\xff"
398
399 julia> escape_string(string('\\u2135','\\0')) # unambiguous
400 "ℵ\\\\0"
401
402 julia> escape_string(string('\\u2135','\\0','0')) # \\0 would be ambiguous
403 "ℵ\\\\x000"
404 ```
405 """
406 function escape_string(io::IO, s::AbstractString, esc=""; keep = ())
407 a = Iterators.Stateful(s)
408 for c::AbstractChar in a
409 if c in esc
410 print(io, '\\', c)
411 elseif c in keep
412 print(io, c)
413 elseif isascii(c)
414 c == '\0' ? print(io, escape_nul(peek(a)::Union{AbstractChar,Nothing})) :
415 c == '\e' ? print(io, "\\e") :
416 c == '\\' ? print(io, "\\\\") :
417 '\a' <= c <= '\r' ? print(io, '\\', "abtnvfr"[Int(c)-6]) :
418 isprint(c) ? print(io, c) :
419 print(io, "\\x", string(UInt32(c), base = 16, pad = 2))
420 elseif !isoverlong(c) && !ismalformed(c)
421 isprint(c) ? print(io, c) :
422 c <= '\x7f' ? print(io, "\\x", string(UInt32(c), base = 16, pad = 2)) :
423 c <= '\uffff' ? print(io, "\\u", string(UInt32(c), base = 16, pad = need_full_hex(peek(a)::Union{AbstractChar,Nothing}) ? 4 : 2)) :
424 print(io, "\\U", string(UInt32(c), base = 16, pad = need_full_hex(peek(a)::Union{AbstractChar,Nothing}) ? 8 : 4))
425 else # malformed or overlong
426 u = bswap(reinterpret(UInt32, c)::UInt32)
427 while true
428 print(io, "\\x", string(u % UInt8, base = 16, pad = 2))
429 (u >>= 8) == 0 && break
430 end
431 end
432 end
433 end

448 unescape_string(str::AbstractString, keep = ())::AbstractString
449 unescape_string(io, s::AbstractString, keep = ())::Nothing
450
451 General unescaping of traditional C and Unicode escape sequences. The first form returns
452 the escaped string, the second prints the result to `io`.
453 The argument `keep` specifies a collection of characters which (along with backlashes) are
454 to be kept as they are.
455
456 The following escape sequences are recognised:
457 - Escaped backslash (`\\\\`)
458 - Escaped double-quote (`\\\"`)
459 - Standard C escape sequences (`\\a`, `\\b`, `\\t`, `\\n`, `\\v`, `\\f`, `\\r`, `\\e`)
460 - Unicode BMP code points (`\\u` with 1-4 trailing hex digits)
461 - All Unicode code points (`\\U` with 1-8 trailing hex digits; max value = 0010ffff)
462 - Hex bytes (`\\x` with 1-2 trailing hex digits)
463 - Octal bytes (`\\` with 1-3 trailing octal digits)
464
465 See also [`escape_string`](@ref).
466
467 # Examples
468 ```jldoctest
469 julia> unescape_string("aaa\\\\nbbb") # C escape sequence
470 "aaa\\nbbb"
471
472 julia> unescape_string("\\\\u03c0") # unicode
473 "π"
474
475 julia> unescape_string("\\\\101") # octal
476 "A"
477
478 julia> unescape_string("aaa \\\\g \\\\n", ['g']) # using `keep` argument
479 "aaa \\\\g \\n"
480 ```
481 """
482 function unescape_string(io::IO, s::AbstractString, keep = ())
483 a = Iterators.Stateful(s)
484 for c in a
485 if !isempty(a) && c == '\\'
486 c = popfirst!(a)
487 if c in keep
488 print(io, '\\', c)
489 elseif c == 'x' || c == 'u' || c == 'U'
490 n = k = 0
491 m = c == 'x' ? 2 :
492 c == 'u' ? 4 : 8
493 while (k += 1) <= m && !isempty(a)
494 nc = peek(a)::AbstractChar
495 n = '0' <= nc <= '9' ? n<<4 + (nc-'0') :
496 'a' <= nc <= 'f' ? n<<4 + (nc-'a'+10) :
497 'A' <= nc <= 'F' ? n<<4 + (nc-'A'+10) : break
498 popfirst!(a)
499 end
500 if k == 1 || n > 0x10ffff
501 u = m == 4 ? 'u' : 'U'
502 throw(ArgumentError("invalid $(m == 2 ? "hex (\\x)" :
503 "unicode (\\$u)") escape sequence"))
504 end
505 if m == 2 # \x escape sequence
506 write(io, UInt8(n))
507 else
508 print(io, Char(n))
509 end
510 elseif '0' <= c <= '7'
511 k = 1
512 n = c-'0'
513 while (k += 1) <= 3 && !isempty(a)
514 c = peek(a)::AbstractChar
515 n = ('0' <= c <= '7') ? n<<3 + c-'0' : break
516 popfirst!(a)
517 end
518 if n > 255
519 throw(ArgumentError("octal escape sequence out of range"))
520 end
521 write(io, UInt8(n))
522 else
523 print(io, c == 'a' ? '\a' :
524 c == 'b' ? '\b' :
525 c == 't' ? '\t' :
526 c == 'n' ? '\n' :
527 c == 'v' ? '\v' :
528 c == 'f' ? '\f' :
529 c == 'r' ? '\r' :
530 c == 'e' ? '\e' :
531 (c == '\\' || c == '"') ? c :
532 throw(ArgumentError("invalid escape sequence \\$c")))
533 end
534 else
535 print(io, c)
536 end
537 end

592 """
593 escape_raw_string(s::AbstractString)
594 escape_raw_string(io, s::AbstractString)
595
596 Escape a string in the manner used for parsing raw string literals.
597 For each double-quote (`"`) character in input string `s`, this
598 function counts the number _n_ of preceding backslash (`\\`) characters,
599 and then increases there the number of backslashes from _n_ to 2_n_+1
600 (even for _n_ = 0). It also doubles a sequence of backslashes at the end
601 of the string.
602
603 This escaping convention is used in raw strings and other non-standard
604 string literals. (It also happens to be the escaping convention
605 expected by the Microsoft C/C++ compiler runtime when it parses a
606 command-line string into the argv[] array.)
607
608 See also [`escape_string`](@ref).
609 """
610 function escape_raw_string(io, str::AbstractString)
611 escapes = 0
612 for c in str
613 if c == '\\'
614 escapes += 1
615 else
616 if c == '"'
617 # if one or more backslashes are followed by
618 # a double quote then escape all backslashes
619 # and the double quote
620 escapes = escapes * 2 + 1
621 end
622 while escapes > 0
623 write(io, '\\')
624 escapes -= 1
625 end
626 escapes = 0
627 write(io, c)
628 end
629 end
630 # also escape any trailing backslashes,
631 # so they do not affect the closing quote
632 while escapes > 0
633 write(io, '\\')
634 write(io, '\\')
635 escapes -= 1
636 end
637 end

675 """
676 unindent(str::AbstractString, indent::Int; tabwidth=8)
677
678 Remove leading indentation from string.
679
680 See also `indent` from the [`MultilineStrings` package](https://github.com/invenia/MultilineStrings.jl).
681
682 # Examples
683 ```jldoctest
684 julia> Base.unindent(" a\\n b", 2)
685 " a\\n b"
686
687 julia> Base.unindent("\\ta\\n\\tb", 2, tabwidth=8)
688 " a\\n b"
689 ```
690 """
691 function unindent(str::AbstractString, indent::Int; tabwidth=8)
692 indent == 0 && return str
693 # Note: this loses the type of the original string
694 buf = IOBuffer(sizehint=sizeof(str))
695 cutting = true
696 col = 0 # current column (0 based)
697 for ch in str
698 if cutting
699 if ch == ' '
700 col += 1
701 elseif ch == '\t'
702 col = div(col + tabwidth, tabwidth) * tabwidth
703 elseif ch == '\n'
704 # Now we need to output enough indentation
705 for i = 1:col-indent
706 print(buf, ' ')
707 end
708 col = 0
709 print(buf, '\n')
710 else
711 cutting = false
712 # Now we need to output enough indentation to get to
713 # correct place
714 for i = 1:col-indent
715 print(buf, ' ')
716 end
717 col += 1
718 print(buf, ch)
719 end
720 elseif ch == '\t' # Handle internal tabs
721 upd = div(col + tabwidth, tabwidth) * tabwidth
722 # output the number of spaces that would have been seen
723 # with original indentation
724 for i = 1:(upd-col)
725 print(buf, ' ')
726 end
727 col = upd
728 elseif ch == '\n'
729 cutting = true
730 col = 0
731 print(buf, '\n')
732 else
733 col += 1
734 print(buf, ch)
735 end
736 end
737 # If we were still "cutting" when we hit the end of the string,
738 # we need to output the right number of spaces for the indentation
739 if cutting
740 for i = 1:col-indent
741 print(buf, ' ')
742 end
743 end
744 String(take!(buf))
745 end
