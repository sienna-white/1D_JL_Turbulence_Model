492
493 Values can be interpolated into `@async` via `\$`, which copies the value directly into the
494 constructed underlying closure. This allows you to insert the _value_ of a variable,

613 function _lift_one_interp_helper(expr::Expr, in_quote_context, letargs)
614 if expr.head === :$

774 # Sticky tasks go into their thread's work queue.
775 if t.sticky
776 tid = Threads.threadid(t)
777 if tid == 0
778 # The task is not yet stuck to a thread. Stick it to the current
779 # thread and do the same to the parent task (the current task) so
780 # that the tasks are correctly co-scheduled (issue #41324).
781 # XXX: Ideally we would be able to unset this.
782 if GC.in_finalizer()
783 # The task was launched in a finalizer. There is no thread to sticky it
784 # to, so just allow it to run anywhere as if it had been non-sticky.
785 t.sticky = false
786 @goto not_sticky
787 else
788 tid = Threads.threadid()
789 ccall(:jl_set_task_tid, Cint, (Any, Cint), t, tid-1)
790 current_task().sticky = true
791 end
792 end
793 push!(workqueue_for(tid), t)
794 else
795 @label not_sticky
796 tp = Threads.threadpool(t)
797 if tp === :foreign || Threads.threadpoolsize(tp) == 1
798 # There's only one thread in the task's assigned thread pool;
799 # use its work queue.
800 tid = (tp === :interactive) ? 1 : Threads.threadpoolsize(:interactive)+1
801 ccall(:jl_set_task_tid, Cint, (Any, Cint), t, tid-1)
802 push!(workqueue_for(tid), t)
803 else
804 # Otherwise, put the task in the multiqueue.
805 Partr.multiq_insert(t, t.priority)
806 tid = 0
807 end
808 end
809 ccall(:jl_wakeup_thread, Cvoid, (Int16,), (tid - 1) % Int16)
810 return t
811 end
812
813 schedule(t::Task) = enq_work(t)
814
815 """
816 schedule(t::Task, [val]; error=false)
817
818 Add a [`Task`](@ref) to the scheduler's queue. This causes the task to run constantly when the system
819 is otherwise idle, unless the task performs a blocking operation such as [`wait`](@ref).
820
821 If a second argument `val` is provided, it will be passed to the task (via the return value of
822 [`yieldto`](@ref)) when it runs again. If `error` is `true`, the value is raised as an exception in
823 the woken task.
824
825 !!! warning
826 It is incorrect to use `schedule` on an arbitrary `Task` that has already been started.
827 See [the API reference](@ref low-level-schedule-wait) for more information.
828
829 # Examples
830 ```jldoctest
831 julia> a5() = sum(i for i in 1:1000);
832
833 julia> b = Task(a5);
834
835 julia> istaskstarted(b)
836 false
837
838 julia> schedule(b);
839
840 julia> yield();
841
842 julia> istaskstarted(b)
843 true
844
845 julia> istaskdone(b)
846 true
847 ```
848 """
849 function schedule(t::Task, @nospecialize(arg); error=false)
850 # schedule a task to be (re)started with the given value or exception
851 t._state === task_state_runnable || Base.error("schedule: Task not runnable")
852 if error
853 t.queue === nothing || Base.list_deletefirst!(t.queue::IntrusiveLinkedList{Task}, t)
854 setfield!(t, :result, arg)
855 setfield!(t, :_isexception, true)
856 else
857 t.queue === nothing || Base.error("schedule: Task not runnable")
858 setfield!(t, :result, arg)
859 end
860 enq_work(t)
861 return t
862 end
863
864 """
865 yield()
866
867 Switch to the scheduler to allow another scheduled task to run. A task that calls this
868 function is still runnable, and will be restarted immediately if there are no other runnable
869 tasks.
870 """
871 function yield()
872 ct = current_task()
873 enq_work(ct)
874 try
875 wait()
876 catch
877 ct.queue === nothing || list_deletefirst!(ct.queue::IntrusiveLinkedList{Task}, ct)
878 rethrow()
879 end
880 end
881
882 @inline set_next_task(t::Task) = ccall(:jl_set_next_task, Cvoid, (Any,), t)
883
884 """
885 yield(t::Task, arg = nothing)
886
887 A fast, unfair-scheduling version of `schedule(t, arg); yield()` which
888 immediately yields to `t` before calling the scheduler.
889 """
890 function yield(t::Task, @nospecialize(x=nothing))
891 (t._state === task_state_runnable && t.queue === nothing) || error("yield: Task not runnable")
892 t.result = x
893 enq_work(current_task())
894 set_next_task(t)
895 return try_yieldto(ensure_rescheduled)
896 end
897
898 """
899 yieldto(t::Task, arg = nothing)
900
901 Switch to the given task. The first time a task is switched to, the task's function is
902 called with no arguments. On subsequent switches, `arg` is returned from the task's last
903 call to `yieldto`. This is a low-level call that only switches tasks, not considering states
904 or scheduling in any way. Its use is discouraged.
905 """
906 function yieldto(t::Task, @nospecialize(x=nothing))
907 # TODO: these are legacy behaviors; these should perhaps be a scheduler
908 # state error instead.
909 if t._state === task_state_done
910 return x
911 elseif t._state === task_state_failed
912 throw(t.result)
913 end
914 t.result = x
915 set_next_task(t)
916 return try_yieldto(identity)
917 end
918
919 function try_yieldto(undo)
920 try
921 ccall(:jl_switch, Cvoid, ())
922 catch
923 undo(ccall(:jl_get_next_task, Ref{Task}, ()))
924 rethrow()
925 end
926 ct = current_task()
927 if ct._isexception
928 exc = ct.result
929 ct.result = nothing
930 ct._isexception = false
931 throw(exc)
932 end
933 result = ct.result
934 ct.result = nothing
935 return result
936 end
937
938 # yield to a task, throwing an exception in it
939 function throwto(t::Task, @nospecialize exc)
940 t.result = exc
941 t._isexception = true
942 set_next_task(t)
943 return try_yieldto(identity)
944 end
945
946 function ensure_rescheduled(othertask::Task)
947 ct = current_task()
948 W = workqueue_for(Threads.threadid())
949 if ct !== othertask && othertask._state === task_state_runnable
950 # we failed to yield to othertask
951 # return it to the head of a queue to be retried later
952 tid = Threads.threadid(othertask)
953 Wother = tid == 0 ? W : workqueue_for(tid)
954 pushfirst!(Wother, othertask)
955 end
956 # if the current task was queued,
957 # also need to return it to the runnable state
958 # before throwing an error
959 list_deletefirst!(W, ct)
960 nothing
961 end
962
963 function trypoptask(W::StickyWorkqueue)
964 while !isempty(W)
965 t = popfirst!(W)
966 if t._state !== task_state_runnable
967 # assume this somehow got queued twice,
968 # probably broken now, but try discarding this switch and keep going
969 # can't throw here, because it's probably not the fault of the caller to wait
970 # and don't want to use print() here, because that may try to incur a task switch
971 ccall(:jl_safe_printf, Cvoid, (Ptr{UInt8}, Int32...),
972 "\nWARNING: Workqueue inconsistency detected: popfirst!(Workqueue).state !== :runnable\n")
973 continue
974 end
975 return t
976 end
977 return Partr.multiq_deletemin()
978 end
979
980 checktaskempty = Partr.multiq_check_empty
981
982 @noinline function poptask(W::StickyWorkqueue)
983 task = trypoptask(W)
984 if !(task isa Task)
985 task = ccall(:jl_task_get_next, Ref{Task}, (Any, Any, Any), trypoptask, W, checktaskempty)
986 end
987 set_next_task(task)
988 nothing
989 end
990
991 function wait()
992 GC.safepoint()
993 W = workqueue_for(Threads.threadid())
994 poptask(W)
995 result = try_yieldto(ensure_rescheduled)
996 process_events()
997 # return when we come out of the queue
998 return result
999 end
