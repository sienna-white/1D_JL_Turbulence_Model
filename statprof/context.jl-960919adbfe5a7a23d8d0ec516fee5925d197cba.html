1 """
2 Internal structure used to track information for a single column in a delimited file.
3
4 Fields:
5 * `type`: always a single, concrete type; no Union{T, Missing}; missingness is tracked in anymissing field; this field is mutable; it may start as one type and get "promoted" to another while parsing; two special types exist: `NeedsTypeDetection`, which specifies that we need to try and detect what type this column's values are and `HardMissing` which means the column type is definitely `Missing` and we don't need to detect anything; to get the "final type" of a column after parsing, call `CSV.coltype(col)`, which takes into account `anymissing`
6 * `anymissing`: whether any missing values have been encountered while parsing; if a user provided a type like `Union{Int, Missing}`, we'll set this to `true`, or when `missing` values are encountered while parsing
7 * `userprovidedtype`: whether the column type was provided by the user or not; this affects whether we'll promote a column's type while parsing, or emit a warning/error depending on `strict` keyword arg
8 * `willdrop`: whether we'll drop this column from the final columnset; computed from select/drop keyword arguments; this will result in a column type of `HardMissing` while parsing, where an efficient parser is used to "skip" a field w/o allocating any parsed value
9 * `pool`: computed from `pool` keyword argument; `true` is `1.0`, `false` is `0.0`, everything else is `Float64(pool)`; once computed, this field isn't mutated at all while parsing; it's used in type detection to determine whether a column will be pooled or not once a type is detected;
10 * `columnspecificpool`: if `pool` was provided via Vector or Dict by user, then `true`, other `false`; if `false`, then only string column types will attempt pooling
11 * `column`: the actual column vector to hold parsed values; field is typed as `AbstractVector` and while parsing, we do switches on `col.type` to assert the column type to make code concretely typed
12 * `lock`: in multithreaded parsing, we have a top-level set of `Vector{Column}`, then each threaded parsing task makes its own copy to parse its own chunk; when synchronizing column types/pooled refs, the task-local `Column` will `lock(col.lock)` to make changes to the parent `Column`; each task-local `Column` shares the same `lock` of the top-level `Column`
13 * `position`: for transposed reading, the current column position
14 * `endposition`: for transposed reading, the expected ending position for this column
15 """

566 elseif select isa Base.Callable
567 for i = 1:ncols
568 select(i, names[i])::Bool || willdrop!(columns, i)
569 end
570 else
571 throw(ArgumentError("`select` keyword argument must be an `AbstractVector` of `Int`, `Symbol`, `String`, or `Bool`, or a selector function of the form `(i, name) -> keep::Bool`"))
572 end
573 elseif drop !== nothing
574 if drop isa AbstractVector{Bool}
575 for i = 1:ncols
576 drop[i] && willdrop!(columns, i)
577 end
578 elseif drop isa AbstractVector{<:Integer}
579 for i = 1:ncols
580 i in drop && willdrop!(columns, i)
581 end
582 elseif drop isa AbstractVector{Symbol} || drop isa AbstractVector{<:AbstractString}
583 drop = map(Symbol, drop)
584 for i = 1:ncols
585 names[i] in drop && willdrop!(columns, i)
586 end
587 elseif drop isa Base.Callable
588 for i = 1:ncols
589 drop(i, names[i])::Bool && willdrop!(columns, i)
590 end
591 else
592 throw(ArgumentError("`drop` keyword argument must be an `AbstractVector` of `Int`, `Symbol`, `String`, or `Bool`, or a selector function of the form `(i, name) -> keep::Bool`"))
593 end
594 end
595 debug && println("computed types are: $types")
596
597 # determine if we can use threads while parsing
598 limit = something(limit, typemax(Int))
599 minrows = min(limit, rowsguess)
600 nthreads = Int(something(ntasks, Threads.nthreads()))
601 if ntasks === nothing && !streaming && nthreads > 1 && !transpose && minrows > (nthreads * 5) && (minrows * ncols) >= 5_000
602 threaded = true
603 ntasks = nthreads
604 elseif ntasks !== nothing && ntasks > 1
605 threaded = true
606 if transpose
607 @warn "`ntasks > 1` not supported on transposed files"
608 threaded = false
609 ntasks = 1
610 elseif minrows < (nthreads * 5)
611 @warn "`ntasks > 1` but there were not enough estimated rows ($minrows) to justify multithreaded parsing"
612 threaded = false
613 ntasks = 1
614 end
615 else
616 threaded = false
617 ntasks = 1
618 end
619 # attempt to chunk up a file for multithreaded parsing; there's chance we can't figure out how to accurately chunk
620 # due to quoted fields, so threaded might get set to false
621 if threaded
622 # when limiting w/ multithreaded parsing, we try to guess about where in the file the limit row # will be
623 # then adjust our final file len to the end of that row
624 # we add some cushion so we hopefully get the limit row correctly w/o shooting past too far and needing to resize! down
625 # but we also don't guarantee limit will be exact w/ multithreaded parsing
626 origrowsguess = rowsguess
627 if limit !== typemax(Int)
628 limit = Int(limit)
629 limitposguess = ceil(Int, (limit / (origrowsguess * 0.8)) * len)
630 if limitposguess < len
631 newlen = [0, limitposguess, min(limitposguess * 2, len)]
632 findchunkrowstart(newlen, 2, buf, options, typemap, downcast, ncols, 5, columns, Type[col.type for col in columns], ReentrantLock(), stringtype, Threads.Atomic{Int}(0), Threads.Atomic{Int}(0), Threads.Atomic{Bool}(true))
633 len = newlen[2] - 1
634 reinitialize_column_type!(columns, types, names, stringtype, streaming)
635 origrowsguess = limit
636 end
637 debug && println("limiting, adjusting len to $len")
638 end
639 chunksize = div(len - datapos, ntasks)
640 chunkpositions = [datapos + chunksize * i for i in 0:ntasks]
641 chunkpositions[end] = len
642 debug && println("initial byte positions before adjusting for start of rows: $chunkpositions")
643 avgbytesperrow, successfullychunked = findrowstarts!(buf, options, chunkpositions, ncols, columns, stringtype, typemap, downcast, rows_to_check)
644 ntasks = length(chunkpositions) - 1
645 if successfullychunked
646 origbytesperrow = ((len - datapos) / origrowsguess)
647 weightedavgbytesperrow = ceil(Int, avgbytesperrow * ((ntasks - 1) / ntasks) + origbytesperrow * (1 / ntasks))
648 rowsguess = ceil(Int, ((len - datapos) / weightedavgbytesperrow) * 1.01)
649 debug && println("single-threaded estimated rows = $origrowsguess, multi-threaded estimated rows = $rowsguess")
650 debug && println("multi-threaded column types sampled as: $columns")
651 else
652 # The following debug statement is doubled by a loud @warning or @error in parsefilechunk!
653 debug && println("multi-threaded parsing failed! Falling back to single thread, reinitializing column types.")
654 reinitialize_column_type!(columns, types, names, stringtype, streaming)
655 threaded = false # the failing is signaled by having !ctx.threaded && ctx.ntasks > 1
656 end
657 end
658 if !threaded
659 chunkpositions = EMPTY_INT_ARRAY
660 if limit < rowsguess
661 rowsguess = limit
662 end
663 end
664
665 end # @inbounds begin
666 return Context(
667 transpose,
668 getname(source),
669 names,
670 rowsguess,
671 ncols,
672 buf,
673 datapos,
674 len,
675 skipto,
676 options,
677 columns,
678 finalpool,
679 downcast,
680 customtypes,
681 typemap,
682 stringtype,
683 limit,
684 threaded,
685 ntasks,
686 chunkpositions,
687 strict,
688 silencewarnings,
689 maxwarnings,
690 debug,
691 tempfile,
692 streaming,
693 types,
694 )
695 end
