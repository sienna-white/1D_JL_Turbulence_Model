3 # This file implements the Semi-NCA (SNCA) dominator tree construction
4 # described in Georgiadis' PhD thesis [LG05], which itself is a simplification
5 # of the Simple Lenguare-Tarjan (SLT) algorithm [LG79]. This algorithm matches
6 # the algorithm choice in LLVM and seems to be a sweet spot in implementation
7 # simplicity and efficiency.
8 #
9 # This file also implements an extension of SNCA that supports updating the
10 # dominator tree with insertion and deletion of edges in the control flow
11 # graph, described in [GI16] as Dynamic SNCA. DSNCA was chosen over DBS, a
12 # different algorithm which achieves the best overall performance in [GI16],
13 # because it is simpler to understand and implement, performs well with edge
14 # deletions, and is of similar performance overall.
15 #
16 # SNCA works by first computing semidominators, then computing immediate
17 # dominators from them. The semidominator of a node is the node with minimum
18 # preorder number such that there is a semidominator path from it to the node.
19 # A semidominator path is a path in which the preorder numbers of all nodes not
20 # at the endpoints are greater than the preorder number of the last node.
21 # Intuitively, the semidominator approximates the immediate dominator of a node
22 # by taking the path (in the CFG) that gets as close to the root as possible
23 # while avoiding ancestors of the node in the DFS tree.
24 #
25 # In computing the semidominators, SNCA performs "path compression" whenever a
26 # node has a nontrivial semidominator (i.e. a semidominator that is not just
27 # its parent in the DFS tree). Path compression propagates the "label" of a
28 # node, which represents a possible semidominator with associated semidominator
29 # path passing through that node.
30 #
31 # For example, path compression will be performed for the following CFG, where
32 # the edge not in the DFS tree is marked with asterisks. Note that nodes are
33 # labeled with their preorder numbers, and all edges point downward.
34 #
35 # 1
36 # |\
37 # | \
38 # | 4
39 # | |
40 # 2 5
41 # | |
42 # | 6
43 # | *
44 # |*
45 # 3
46 #
47 # There is a nontrivial semidominator path from 1 to 3, passing through 4, 5,
48 # and 6. Stepping through the whole algorithm on paper with an example like
49 # this is very helpful for understanding how it works.
50 #
51 # DSNCA runs the whole algorithm from scratch if the DFS tree is invalidated by
52 # the insertion or deletion, but otherwise recomputes a subset of the
53 # semidominators (all immediate dominators then need to be recomputed).
54 #
55 # [LG05] Linear-Time Algorithms for Dominators and Related Problems
56 # Loukas Georgiadis, Princeton University, November 2005, pp. 21-23:
57 # ftp://ftp.cs.princeton.edu/reports/2005/737.pdf
58 #
59 # [LT79] A fast algorithm for finding dominators in a flowgraph
60 # Thomas Lengauer, Robert Endre Tarjan, July 1979, ACM TOPLAS 1-1
61 # http://www.dtic.mil/dtic/tr/fulltext/u2/a054144.pdf
62 #
63 # [GI16] An Experimental Study of Dynamic Dominators
64 # Loukas Georgiadis, Giuseppe F. Italiano, Luigi Laura, Federico
65 # Santaroni, April 2016
66 # https://arxiv.org/abs/1604.02711

500 """
501 Check if x is on some semidominator path from the semidominator of y to y,
502 assuming there is an edge from x to y.
503 """
504 function on_semidominator_path(domtree::DomTree, x::BBNumber, y::BBNumber)
505 x_pre = domtree.dfs_tree.to_pre[x]
506 y_pre = domtree.dfs_tree.to_pre[y]
507
508 semi_y = domtree.snca_state[y_pre].semi
509 current_block = x_pre
510
511 # Follow the semidominators of `x` up the DFS tree to see if we ever reach
512 # the semidominator of `y`. If so, `x` is on a semidominator path between
513 # `y` and its semidominator. We can stop if the preorder number of the
514 # semidominators becomes less than that of the semidominator of `y`,
515 # because it can only decrease further.
516 while current_block >= semi_y
517 if semi_y == current_block
518 return true
519 end
520 current_block = domtree.snca_state[current_block].semi
521 end
522 return false

542 domtree.nodes[new_bb] = old_nodes[old_bb]
543 map!(i -> rename_bb[i],
544 domtree.nodes[new_bb].children,
545 domtree.nodes[new_bb].children)
546 end
547 end
548
549 # length of `to_pre` after renaming DFS tree is new number of basic blocks
550 resize!(domtree.idoms_bb, length(domtree.dfs_tree.to_pre))
551 resize!(domtree.nodes, length(domtree.dfs_tree.to_pre))
552 return domtree
553 end

555 """
556 Rename basic block numbers in a DFS tree, removing the block if it is renamed
557 to -1.
558 """
559 function rename_nodes!(D::DFSTree, rename_bb::Vector{BBNumber})
560 n_blocks = length(D.to_pre)
561 n_reachable_blocks = length(D.from_pre)
562
563 old_to_pre = copy(D.to_pre)
564 old_from_pre = copy(D.from_pre)
565 old_to_post = copy(D.to_post)
566 old_from_post = copy(D.from_post)
567 max_new_bb = 0
568 for (old_bb, new_bb) in enumerate(rename_bb)
569 if new_bb != -1
570 D.to_pre[new_bb] = old_to_pre[old_bb]
571 D.from_pre[old_to_pre[old_bb]] = new_bb
572 D.to_post[new_bb] = old_to_post[old_bb]
573 D.from_post[old_to_post[old_bb]] = new_bb
574
575 # Keep track of highest BB number to resize arrays with
576 if new_bb > max_new_bb
577 max_new_bb = new_bb
578 end
579 end
580 end
581 resize!(D.to_pre, max_new_bb)
582 resize!(D.to_post, max_new_bb)
583 # `to_parent_pre` should be unchanged
584 return D
585 end

587 """
588 dominates(domtree::DomTree, bb1::Int, bb2::Int) -> Bool
589
590 Checks if `bb1` dominates `bb2`.
591 `bb1` and `bb2` are indexes into the `CFG` blocks.
592 `bb1` dominates `bb2` if the only way to enter `bb2` is via `bb1`.
593 (Other blocks may be in between, e.g `bb1->bbx->bb2`).
594 """
595 dominates(domtree::DomTree, bb1::BBNumber, bb2::BBNumber) =
596 _dominates(domtree, bb1, bb2)
597
598 """
599 postdominates(domtree::DomTree, bb1::Int, bb2::Int) -> Bool
600
601 Checks if `bb1` post-dominates `bb2`.
602 `bb1` and `bb2` are indexes into the `CFG` blocks.
603 `bb1` post-dominates `bb2` if every pass from `bb2` to the exit is via `bb1`.
604 (Other blocks may be in between, e.g `bb2->bbx->bb1->exit`).
605 """
606 postdominates(domtree::PostDomTree, bb1::BBNumber, bb2::BBNumber) =
607 _dominates(domtree, bb1, bb2)
608
609 function _dominates(domtree::GenericDomTree, bb1::BBNumber, bb2::BBNumber)
610 bb1 == bb2 && return true
611 target_level = domtree.nodes[bb1].level
612 source_level = domtree.nodes[bb2].level
613 source_level < target_level && return false
614 for _ in (source_level - 1):-1:target_level
615 bb2 = domtree.idoms_bb[bb2]
616 end
617 return bb1 == bb2
618 end
619
620 bb_unreachable(domtree::DomTree, bb::BBNumber) = bb != 1 && domtree.dfs_tree.to_pre[bb] == 0
621
622 "Iterable data structure that walks though all dominated blocks"
623 struct DominatedBlocks
624 domtree::DomTree
625 worklist::Vector{BBNumber}
626 end
627
628 "Returns an iterator that walks through all blocks dominated by the basic block at index `root`"
629 function dominated(domtree::DomTree, root::BBNumber)
630 doms = DominatedBlocks(domtree, Vector{BBNumber}())
631 push!(doms.worklist, root)
632 doms
633 end
634
635 function iterate(doms::DominatedBlocks, state::Nothing=nothing)
636 isempty(doms.worklist) && return nothing
637 bb = pop!(doms.worklist)
638 for dominated in doms.domtree.nodes[bb].children
639 push!(doms.worklist, dominated)
640 end
641 return (bb, nothing)
642 end

644 """
645 nearest_common_dominator(domtree::GenericDomTree, a::BBNumber, b::BBNumber)
646
647 Compute the nearest common (post-)dominator of `a` and `b`.
648 """
649 function nearest_common_dominator(domtree::GenericDomTree, a::BBNumber, b::BBNumber)
650 alevel = domtree.nodes[a].level
651 blevel = domtree.nodes[b].level
652 # W.l.g. assume blevel <= alevel
653 if alevel < blevel
654 a, b = b, a
655 alevel, blevel = blevel, alevel
656 end
657 while alevel > blevel
658 a = domtree.idoms_bb[a]
659 alevel -= 1
660 end
661 while a != b && a != 0
662 a = domtree.idoms_bb[a]
663 b = domtree.idoms_bb[b]
664 end
665 @assert a == b
666 return a
667 end
668
669 function naive_idoms(blocks::Vector{BasicBlock}, is_post_dominator::Bool=false)
670 nblocks = length(blocks)
671 # The extra +1 helps us detect unreachable blocks below
672 dom_all = BitSet(1:nblocks+1)
673 dominators = is_post_dominator ?
674 BitSet[isempty(blocks[n].succs) ? BitSet(n) : copy(dom_all) for n = 1:nblocks] :
675 BitSet[n == 1 ? BitSet(1) : copy(dom_all) for n = 1:nblocks]
676 changed = true
677 relevant_blocks = (is_post_dominator ? (1:nblocks) : (2:nblocks))
678 while changed
679 changed = false
680 for n in relevant_blocks
681 edges = is_post_dominator ? blocks[n].succs : blocks[n].preds
682 if isempty(edges)
683 continue
684 end
685 firstp, rest = Iterators.peel(Iterators.filter(p->p != 0, edges))::NTuple{2,Any}
686 new_doms = copy(dominators[firstp])
687 for p in rest
688 intersect!(new_doms, dominators[p])
689 end
690 push!(new_doms, n)
691 changed = changed || (new_doms != dominators[n])
692 dominators[n] = new_doms
693 end
694 end
695 # Compute idoms
696 idoms = fill(0, nblocks)
697 for i in relevant_blocks
698 if dominators[i] == dom_all
699 idoms[i] = 0
700 continue
701 end
702 doms = collect(dominators[i])
703 for dom in doms
704 i == dom && continue
705 hasany = false
706 for p in doms
707 if p !== i && p !== dom && dom in dominators[p]
708 hasany = true; break
709 end
710 end
711 hasany && continue
712 idoms[i] = dom
713 end
714 end
715 idoms
716 end
