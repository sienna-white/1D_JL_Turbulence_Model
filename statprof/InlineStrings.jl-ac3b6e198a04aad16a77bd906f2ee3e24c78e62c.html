577 # Only benefit from keeping the small-ish strings as InlineStrings
578 function _string(a::Ta, b::Tb) where {Ta <: SmallInlineStrings, Tb <: SmallInlineStrings}
579 T = summed_type(Ta, Tb)
580 len_a = sizeof(a)
581 len_b = sizeof(b)
582 # Remove length byte (lshr), grow to new size (zext), move chars forward (shl).
583 a2 = Base.shl_int(Base.zext_int(T, Base.lshr_int(a, 8)), 8 * (sizeof(T) - sizeof(Ta) + 1))
584 b2 = Base.shl_int(Base.zext_int(T, Base.lshr_int(b, 8)), 8 * (sizeof(T) - sizeof(Tb) + 1 - len_a))
585 lb = _oftype(T, len_a + len_b) # new length byte
586 return Base.or_int(Base.or_int(a2, b2), lb)
587 end

601 function Base.repeat(x::T, r::Integer) where {T <: InlineString}
602 r < 0 && throw(ArgumentError("can't repeat a string $r times"))
603 r == 0 && return ""
604 r == 1 && return s
605 n = sizeof(x)
606 out = Base._string_n(n * r)
607 if n == 1 # common case: repeating a single-byte string
608 @inbounds b = codeunit(x, 1)
609 ccall(:memset, Ptr{Cvoid}, (Ptr{UInt8}, Cint, Csize_t), out, b, r)
610 else
611 for i = 0:r-1
612 ref = Ref{T}(_bswap(x))
613 GC.@preserve ref out begin
614 ptr = convert(Ptr{UInt8}, Base.unsafe_convert(Ptr{T}, ref))
615 unsafe_copyto!(pointer(out, i * n + 1), ptr, n)
616 end
617 end
618 end
619 return out
620 end

666 Base.@propagate_inbounds function Base.thisind(s::InlineString, i::Int)
667 i == 0 && return 0
668 n = ncodeunits(s)
669 i == n + 1 && return i
670 @boundscheck Base.between(i, 1, n) || throw(BoundsError(s, i))
671 @inbounds b = codeunit(s, i)
672 (b & 0xc0 == 0x80) & (i-1 > 0) || return i
673 @inbounds b = codeunit(s, i-1)
674 Base.between(b, 0b11000000, 0b11110111) && return i-1
675 (b & 0xc0 == 0x80) & (i-2 > 0) || return i
676 @inbounds b = codeunit(s, i-2)
677 Base.between(b, 0b11100000, 0b11110111) && return i-2
678 (b & 0xc0 == 0x80) & (i-3 > 0) || return i
679 @inbounds b = codeunit(s, i-3)
680 Base.between(b, 0b11110000, 0b11110111) && return i-3
681 return i
682 end

684 Base.@propagate_inbounds function Base.nextind(s::InlineString, i::Int)
685 i == 0 && return 1
686 n = ncodeunits(s)
687 @boundscheck Base.between(i, 1, n) || throw(BoundsError(s, i))
688 @inbounds l = codeunit(s, i)
689 (l < 0x80) | (0xf8 ≤ l) && return i+1
690 if l < 0xc0
691 i′ = @inbounds thisind(s, i)
692 return i′ < i ? @inbounds(nextind(s, i′)) : i+1
693 end
694 # first continuation byte
695 (i += 1) > n && return i
696 @inbounds b = codeunit(s, i)
697 b & 0xc0 ≠ 0x80 && return i
698 ((i += 1) > n) | (l < 0xe0) && return i
699 # second continuation byte
700 @inbounds b = codeunit(s, i)
701 b & 0xc0 ≠ 0x80 && return i
702 ((i += 1) > n) | (l < 0xf0) && return i
703 # third continuation byte
704 @inbounds b = codeunit(s, i)
705 ifelse(b & 0xc0 ≠ 0x80, i, i+1)
706 end

716 function iterate_continued(s::InlineString, i::Int, u::UInt32)
717 u < 0xc0000000 && (i += 1; @goto ret)
718 n = ncodeunits(s)
719 # first continuation byte
720 (i += 1) > n && @goto ret
721 @inbounds b = codeunit(s, i)
722 b & 0xc0 == 0x80 || @goto ret
723 u |= UInt32(b) << 16
724 # second continuation byte
725 ((i += 1) > n) | (u < 0xe0000000) && @goto ret
726 @inbounds b = codeunit(s, i)
727 b & 0xc0 == 0x80 || @goto ret
728 u |= UInt32(b) << 8
729 # third continuation byte
730 ((i += 1) > n) | (u < 0xf0000000) && @goto ret
731 @inbounds b = codeunit(s, i)
732 b & 0xc0 == 0x80 || @goto ret
733 u |= UInt32(b); i += 1
734 @label ret
735 return reinterpret(Char, u), i
736 end

785 i < n || return c

787 @inbounds while true
788 while true
789 (i += 1) ≤ n || return c
790 0xc0 ≤ b ≤ 0xf7 && break
791 b = codeunit(s, i)

795 c -= (x = b & 0xc0 == 0x80)

810 using Base.Sort, Base.Order

812 # And under certain thresholds, MergeSort is faster than RadixSort, even for small InlineStrings
813 const MergeSortThresholds = Dict(
814 2 => 2^5,
815 4 => 2^7,
816 8 => 2^9,
817 16 => 2^23
818 )

826 struct Radix
827 size::Int
828 pow::Int
829 mask::UInt16
830 end

832 Radix(size) = Radix(size, 2^size, typemax(UInt16) >> (16 - size))

834 sortvalue(o::By, x ) = sortvalue(Forward, o.by(x))
835 sortvalue(o::Perm, i::Int) = sortvalue(o.order, o.data[i])
836 sortvalue(o::Lt, x ) = error("sortvalue does not work with general Lt Orderings")
837 sortvalue(rev::ReverseOrdering, x) = Base.not_int(sortvalue(rev.fwd, x))
838 sortvalue(::Base.ForwardOrdering, x) = x

840 _oftype(::Type{T}, x::S) where {T, S} = sizeof(T) == sizeof(S) ? Base.bitcast(T, x) : sizeof(T) > sizeof(S) ? Base.zext_int(T, x) : Base.trunc_int(T, x)

842 radix(v::T, j, radix_size, radix_mask) where {T} = _oftype(Int64, Base.and_int(Base.lshr_int(v, (j - 1) * radix_size), _oftype(T, radix_mask))) + 1

844 @noinline requireprimitivetype(T) = throw(ArgumentError("InlineStringSort requires isprimitivetype input: `$T` invalid"))
845
846 function Base.sort!(vs::AbstractVector, lo::Int, hi::Int, ::InlineStringSortAlg, o::Ordering)
847 # Input checking
848 lo >= hi && return vs
849
850 # Make sure we're sorting a primitive type
851 T = Base.Order.ordtype(o, vs)
852 isprimitivetype(Base.nonmissingtype(T)) || requireprimitivetype(T)
853
854 if hi - lo < MergeSortThresholds[sizeof(T)]
855 return sort!(vs, lo, hi, MergeSort, o)
856 end
857
858 # setup
859 ts = similar(vs)
860 rdx = Radix(11)
861 radix_size = rdx.size
862 radix_mask = rdx.mask
863 radix_size_pow = rdx.pow
864 # iters is the # of 11-bit chunks we split each element up into
865 # they each represent a "significant digit" we'll be sorting on
866 iters = cld(sizeof(T) * 8, radix_size)
867 # bin has a row for each unique 11-bit pattern
868 # and a column for each 11-bit chunk we'll split each element up into
869 bin = zeros(UInt32, radix_size_pow, iters)
870 # if for some reason our lo isn't 1, we want to start our
871 # 1st row bin values as the 1st index we'll start at in the output
872 # i.e. we're assuming firstindex(vs):(lo - 1) is already sorted
873 if lo > 1; bin[1, :] .= lo-1; end
874
875 # for each element, split into 11-bit chunks (radix)
876 # and accumulate counts per unique pattern in bin
877 for i = lo:hi
878 v = sortvalue(o, vs[i])
879 for j = 1:iters
880 idx = radix(v, j, radix_size, radix_mask)
881 @inbounds bin[idx, j] += 1
882 end
883 end
884
885 # now we sort elements by sorting each radix using counting sort
886 swaps = 0
887 len = hi - lo + 1
888 @inbounds for j = 1:iters
889 # we first check if the radix for each element happened to be
890 # the exact same bit pattern; if so, they're "already sorted"
891 # for this radix and we can skip to the next. This would be common
892 # if we, for example, had many small integer values stored in Int64
893 # which would result in many "wasted" zero bits in most elements
894 v = sortvalue(o, vs[hi])
895 idx = radix(v, j, radix_size, radix_mask)
896
897 # if every element was counted at this bit pattern
898 # we can skip to the next radix chunk
899 bin[idx, j] == len && continue
900
901 # otherwise, we perform the counting sort for this radix
902 # by doing a cumulative sum for this radix column in bin
903 x = bin[1, j]
904 for i = 2:radix_size_pow
905 x += bin[i, j]
906 bin[i, j] = x
907 end
908 # now we extract the output index for our 1st element (vs[hi])
909 ci = bin[idx, j]
910 # and decrement the count for that bit pattern which
911 # will result in a subsequent identical bit pattern being
912 # placed one index ahead of the current one
913 bin[idx, j] -= 1
914 ts[ci] = vs[hi]
915
916 # now we sort the rest of the elements' radix similarly
917 for i in (hi - 1):-1:lo
918 v = sortvalue(o, vs[i])
919 idx = radix(v, j, radix_size, radix_mask)
920 ci = bin[idx, j]
921 bin[idx, j] -= 1
922 ts[ci] = vs[i]
923 end
924 # we keep 2 arrays, vs and ts
925 # because we can't overwrite where the current
926 # element will go in the output before we've sorted
927 # the element already there
928 vs, ts = ts, vs
929 swaps += 1
930 end
931
932 if isodd(swaps)
933 vs, ts = ts, vs
934 @inbounds for i = lo:hi
935 vs[i] = ts[i]
936 end
937 end
938 return vs
939 end

941 # collections of InlineStrings
942 """
943 inlinestrings(itr) => Vector
944
945 Utility function that takes any iterator of `AbstractString` values
946 and attempts to produce a `Vector` with a single promoted `InlineString` type. That is,
947 all iterated elements will be promoted to the smallest `InlineString` subtype
948 that can fit all elements. If any value is larger than the current largest InlineString
949 type (256 bytes), the entire collection will be promoted to `String` instead.
950 `missing` values are also allowed and will result in a result eltype of `Union{Missing, X}`
951 where `X` is an `InlineString` subtype or `String`.
952 """
953 function inlinestrings(itr::T) where {T}
954 # x must be iterable
955 IS = Base.IteratorSize(T)
956 state = iterate(itr)
957 state === nothing && return []
958 y, st = state
959 x = y === missing ? missing : sizeof(y) < 256 ? InlineString(y) : String(y)
960 eT = typeof(x)
961 # allocate res, which will either be same length as `itr` if
962 # IS <: HasLength, or length of 0 if Base.SizeUnknown
963 res = allocate(eT, IS, itr)
964 i = 1
965 # set! push!-es for Base.SizeUnknown, or setindex! for HasLength
966 set!(IS, res, x, i)
967 i += 1
968 # dispatch to separate function for type stability
969 return _inlinestrings(itr, st, eT, IS, res, i)
970 end
971
972 const HasLength = Union{Base.HasShape, Base.HasLength}
973 allocate(::Type{T}, ::HasLength, itr) where {T} = Vector{T}(undef, length(itr))
974 allocate(::Type{T}, IS, itr) where {T} = Vector{T}(undef, 0)
975 set!(::HasLength, res, x, i) = setindex!(res, x, i)
976 set!(IS, res, x, i) = push!(res, x)
977
978 function _inlinestrings(itr, st, ::Type{eT}, IS, res, i) where {eT}
979 while true
980 state = iterate(itr, st)
981 state === nothing && break
982 y, st = state
983 if y === missing && eT >: Missing
984 set!(IS, res, missing, i)
985 elseif y !== missing && eT !== Missing && (sizeof(y) < sizeof(eT) || sizeof(y) == 1)
986 set!(IS, res, Base.nonmissingtype(eT)(y), i)
987 else
988 # need to promote and widen res,
989 # then re-dispatch on _inlinestrings for new eltype
990 x = y === missing ? missing : sizeof(y) < 256 ? InlineString(y) : String(y)
991 new_eT = promote_type(typeof(x), eT)
992 newres = allocate(new_eT, Base.HasLength(), res)
993 copyto!(newres, 1, res, 1, i - 1)
994 set!(IS, newres, x, i)
995 return _inlinestrings(itr, st, new_eT, IS, newres, i + 1)
996 end
997 i += 1
998 end
999 return res
1000 end
