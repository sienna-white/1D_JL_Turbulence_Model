39 @cfunction(callable, ReturnType, (ArgumentTypes...,)) -> Ptr{Cvoid}
40 @cfunction(\$callable, ReturnType, (ArgumentTypes...,)) -> CFunction
41
42 Generate a C-callable function pointer from the Julia function `callable`
43 for the given type signature.
44 To pass the return value to a `ccall`, use the argument type `Ptr{Cvoid}` in the signature.
45
46 Note that the argument type tuple must be a literal tuple, and not a tuple-valued variable or expression
47 (although it can include a splat expression). And that these arguments will be evaluated in global scope
48 during compile-time (not deferred until runtime).
49 Adding a '\\\$' in front of the function argument changes this to instead create a runtime closure
50 over the local variable `callable` (this is not supported on all architectures).
51

309 function transcode(::Type{UInt16}, src::AbstractVector{UInt8})
310 require_one_based_indexing(src)
311 dst = UInt16[]
312 i, n = 1, length(src)
313 n > 0 || return dst
314 sizehint!(dst, 2n)
315 a = src[1]
316 while true
317 if i < n && -64 <= a % Int8 <= -12 # multi-byte character
318 b = src[i += 1]
319 if -64 <= (b % Int8) || a == 0xf4 && 0x8f < b
320 # invalid UTF-8 (non-continuation or too-high code point)
321 push!(dst, a)
322 a = b; continue
323 elseif a < 0xe0 # 2-byte UTF-8
324 push!(dst, xor(0x3080, UInt16(a) << 6, b))
325 elseif i < n # 3/4-byte character
326 c = src[i += 1]
327 if -64 <= (c % Int8) # invalid UTF-8 (non-continuation)
328 push!(dst, a, b)
329 a = c; continue
330 elseif a < 0xf0 # 3-byte UTF-8
331 push!(dst, xor(0x2080, UInt16(a) << 12, UInt16(b) << 6, c))
332 elseif i < n
333 d = src[i += 1]
334 if -64 <= (d % Int8) # invalid UTF-8 (non-continuation)
335 push!(dst, a, b, c)
336 a = d; continue
337 elseif a == 0xf0 && b < 0x90 # overlong encoding
338 push!(dst, xor(0x2080, UInt16(b) << 12, UInt16(c) << 6, d))
339 else # 4-byte UTF-8
340 push!(dst, 0xe5b8 + (UInt16(a) << 8) + (UInt16(b) << 2) + (c >> 4),
341 xor(0xdc80, UInt16(c & 0xf) << 6, d))
342 end
343 else # too short
344 push!(dst, a, b, c)
345 break
346 end
347 else # too short
348 push!(dst, a, b)
349 break
350 end
351 else # ASCII or invalid UTF-8 (continuation byte or too-high code point)
352 push!(dst, a)
353 end
354 i < n || break
355 a = src[i += 1]
356 end
357 return dst
358 end

360 function transcode(::Type{UInt8}, src::AbstractVector{UInt16})
361 require_one_based_indexing(src)
362 n = length(src)
363 n == 0 && return UInt8[]
364
365 # Precompute m = sizeof(dst). This involves annoying duplication
366 # of the loop over the src array. However, this is not just an
367 # optimization: it is problematic for security reasons to grow
368 # dst dynamically, because Base.winprompt uses this function to
369 # convert passwords to UTF-8 and we don't want to make unintentional
370 # copies of the password data.
371 a = src[1]
372 i, m = 1, 0
373 while true
374 if a < 0x80
375 m += 1
376 elseif a < 0x800 # 2-byte UTF-8
377 m += 2
378 elseif a & 0xfc00 == 0xd800 && i < length(src)
379 b = src[i += 1]
380 if (b & 0xfc00) == 0xdc00 # 2-unit UTF-16 sequence => 4-byte UTF-8
381 m += 4
382 else
383 m += 3
384 a = b; continue
385 end
386 else
387 # 1-unit high UTF-16 or unpaired high surrogate
388 # either way, encode as 3-byte UTF-8 code point
389 m += 3
390 end
391 i < n || break
392 a = src[i += 1]
393 end
394
395 dst = StringVector(m)
396 a = src[1]
397 i, j = 1, 0
398 while true
399 if a < 0x80 # ASCII
400 dst[j += 1] = a % UInt8
401 elseif a < 0x800 # 2-byte UTF-8
402 dst[j += 1] = 0xc0 | ((a >> 6) % UInt8)
403 dst[j += 1] = 0x80 | ((a % UInt8) & 0x3f)
404 elseif a & 0xfc00 == 0xd800 && i < n
405 b = src[i += 1]
406 if (b & 0xfc00) == 0xdc00
407 # 2-unit UTF-16 sequence => 4-byte UTF-8
408 a += 0x2840
409 dst[j += 1] = 0xf0 | ((a >> 8) % UInt8)
410 dst[j += 1] = 0x80 | ((a % UInt8) >> 2)
411 dst[j += 1] = xor(0xf0, ((a % UInt8) << 4) & 0x3f, (b >> 6) % UInt8)
412 dst[j += 1] = 0x80 | ((b % UInt8) & 0x3f)
413 else
414 dst[j += 1] = 0xe0 | ((a >> 12) % UInt8)
415 dst[j += 1] = 0x80 | (((a >> 6) % UInt8) & 0x3f)
416 dst[j += 1] = 0x80 | ((a % UInt8) & 0x3f)
417 a = b; continue
418 end
419 else
420 # 1-unit high UTF-16 or unpaired high surrogate
421 # either way, encode as 3-byte UTF-8 code point
422 dst[j += 1] = 0xe0 | ((a >> 12) % UInt8)
423 dst[j += 1] = 0x80 | (((a >> 6) % UInt8) & 0x3f)
424 dst[j += 1] = 0x80 | ((a % UInt8) & 0x3f)
425 end
426 i < n || break
427 a = src[i += 1]
428 end
429 return dst
430 end

566 ccall_macro_parse(expression)
567
568 `ccall_macro_parse` is an implementation detail of `@ccall`.
569
570 It takes an expression like `:(printf("%d"::Cstring, value::Cuint)::Cvoid)`
571 returns: a tuple of `(function_name, return_type, arg_types, args)`
572
573 The above input outputs this:
574
575 (:printf, :Cvoid, [:Cstring, :Cuint], ["%d", :value])
576 """
577 function ccall_macro_parse(expr::Expr)
578 # setup and check for errors
579 if !Meta.isexpr(expr, :(::))
580 throw(ArgumentError("@ccall needs a function signature with a return type"))
581 end
582 rettype = expr.args[2]
583
584 call = expr.args[1]
585 if !Meta.isexpr(call, :call)
586 throw(ArgumentError("@ccall has to take a function call"))
587 end
588
589 # get the function symbols
590 func = let f = call.args[1]
591 if Meta.isexpr(f, :.)
592 :(($(f.args[2]), $(f.args[1])))
593 elseif Meta.isexpr(f, :$)
594 f
595 elseif f isa Symbol
596 QuoteNode(f)
597 else
598 throw(ArgumentError("@ccall function name must be a symbol, a `.` node (e.g. `libc.printf`) or an interpolated function pointer (with `\$`)"))
599 end
600 end
601
602 # detect varargs
603 varargs = nothing
604 argstart = 2
605 callargs = call.args
606 if length(callargs) >= 2 && Meta.isexpr(callargs[2], :parameters)
607 argstart = 3
608 varargs = callargs[2].args
609 end
610
611 # collect args and types
612 args = []
613 types = []
614
615 function pusharg!(arg)
616 if !Meta.isexpr(arg, :(::))
617 throw(ArgumentError("args in @ccall need type annotations. '$arg' doesn't have one."))
618 end
619 push!(args, arg.args[1])
620 push!(types, arg.args[2])
621 end
622
623 for i in argstart:length(callargs)
624 pusharg!(callargs[i])
625 end
626 # add any varargs if necessary
627 nreq = 0
628 if !isnothing(varargs)
629 if length(args) == 0
630 throw(ArgumentError("C ABI prohibits vararg without one required argument"))
631 end
632 nreq = length(args)
633 for a in varargs
634 pusharg!(a)
635 end
636 end
637
638 return func, rettype, types, args, nreq
639 end

673 literal. The library may be omitted, in which case the `function_name`
674 is resolved in the current process. Alternatively, `@ccall` may
675 also be used to call a function pointer `\$function_pointer`, such as
676 one returned by `dlsym`.
677
678 Each `argvalue` to `@ccall` is converted to the corresponding
679 `argtype`, by automatic insertion of calls to `unsafe_convert(argtype,
680 cconvert(argtype, argvalue))`. (See also the documentation for
681 [`unsafe_convert`](@ref Base.unsafe_convert) and [`cconvert`](@ref
682 Base.cconvert) for further details.) In most cases, this simply
683 results in a call to `convert(argtype, argvalue)`.
684
685
686 # Examples
687
688 @ccall strlen(s::Cstring)::Csize_t
689
690 This calls the C standard library function:
691
692 size_t strlen(char *)
693
694 with a Julia variable named `s`. See also `ccall`.
695
696 Varargs are supported with the following convention:
697
698 @ccall printf("%s = %d"::Cstring ; "foo"::Cstring, foo::Cint)::Cint
699
700 The semicolon is used to separate required arguments (of which there
701 must be at least one) from variadic arguments.
702
703 Example using an external library:
704
705 # C signature of g_uri_escape_string:
706 # char *g_uri_escape_string(const char *unescaped, const char *reserved_chars_allowed, gboolean allow_utf8);
707
708 const glib = "libglib-2.0"
709 @ccall glib.g_uri_escape_string(my_uri::Cstring, ":/"::Cstring, true::Cint)::Cstring
710
711 The string literal could also be used directly before the function
712 name, if desired `"libglib-2.0".g_uri_escape_string(...`
713 """
714 macro ccall(expr)
715 return ccall_macro_lower(:ccall, ccall_macro_parse(expr)...)
716 end
