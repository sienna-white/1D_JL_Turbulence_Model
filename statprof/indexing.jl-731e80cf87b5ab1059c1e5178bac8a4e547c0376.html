247 function _need_batch(chunks, i, batch_strategy)
248 needsbatch, chunksrem = _need_batch_index(first(i), chunks, batch_strategy)
249 needsbatch ? true : _need_batch(chunksrem, Base.tail(i), batch_strategy)
250 end

254
255 # Integer,UnitRange and Colon are contiguous and dont need batching
256 _need_batch_index(::Union{Integer,UnitRange,Colon}, chunks, _) = false, Base.tail(chunks)
257 # CartesianIndices are contiguous, also dont need batching, but chunks need splitting
258 _need_batch_index(i::CartesianIndices{N}, chunks, _) where {N} = false, last(splitchunks(i, chunks))

262 _need_batch_index(::StepRange, chunks, ::BatchStrategy{CanStepRange}) = false, Base.tail(chunks)
263 # Everything else may need batching
264 function _need_batch_index(i, chunks, batchstrategy)
265 chunksnow, chunksrem = splitchunks(i, chunks)
266 allow_multi = allow_multi_chunk_access(batchstrategy)
267 needsbatch = (allow_multi || has_chunk_gap(approx_chunksize.(chunksnow), i)) &&
268 is_sparse_index(i; density_threshold=density_threshold(batchstrategy))
