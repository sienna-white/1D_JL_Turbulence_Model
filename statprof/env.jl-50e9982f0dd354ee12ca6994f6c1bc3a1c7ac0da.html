9 function memoized_env_lookup(str::AbstractString)
10 # Windows environment variables have a different format from Linux / MacOS, and previously
11 # incurred allocations because we had to convert a String to a Vector{Cwchar_t} each time
12 # an environment variable was looked up. This function memoizes that lookup process, storing
13 # the String => Vector{Cwchar_t} pairs in env_dict
14 @lock env_lock begin
15 var = get(env_dict, str, nothing)
16 if isnothing(var)
17 var = cwstring(str)
18 env_dict[str] = var
19 end
20 return var
21 end
22 end

84 """
85 ENV
86
87 Reference to the singleton `EnvDict`, providing a dictionary interface to system environment
88 variables.
89
90 (On Windows, system environment variables are case-insensitive, and `ENV` correspondingly converts
91 all keys to uppercase for display, iteration, and copying. Portable code should not rely on the
92 ability to distinguish variables by case, and should beware that setting an ostensibly lowercase
93 variable may result in an uppercase `ENV` key.)
94
95 !!! warning
96 Mutating the environment is not thread-safe.
97
98 # Examples
99 ```julia-repl
100 julia> ENV
101 Base.EnvDict with "50" entries:
102 "SECURITYSESSIONID" => "123"
103 "USER" => "username"
104 "MallocNanoZone" => "0"
105 ⋮ => ⋮
106
107 julia> ENV["JULIA_EDITOR"] = "vim"
108 "vim"
109
110 julia> ENV["JULIA_EDITOR"]
111 "vim"
112 ```
113
114 See also: [`withenv`](@ref), [`addenv`](@ref).
115 """

131 """
132 Base.get_bool_env(name::String, default::Bool)::Union{Bool,Nothing}
133
134 Evaluate whether the value of environnment variable `name` is a truthy or falsy string,
135 and return `nothing` if it is not recognized as either. If the variable is not set, or is set to "",
136 return `default`.
137
138 Recognized values are the following, and their Capitalized and UPPERCASE forms:
139 truthy: "t", "true", "y", "yes", "1"
140 falsy: "f", "false", "n", "no", "0"
141 """
142 function get_bool_env(name::String, default::Bool)
143 haskey(ENV, name) || return default
144 val = ENV[name]
145 if isempty(val)
146 return default
147 elseif val in get_bool_env_truthy
148 return true
149 elseif val in get_bool_env_falsy
150 return false
151 else
152 return nothing
153 end
154 end

238 """
239 withenv(f, kv::Pair...)
240
241 Execute `f` in an environment that is temporarily modified (not replaced as in `setenv`)
242 by zero or more `"var"=>val` arguments `kv`. `withenv` is generally used via the
243 `withenv(kv...) do ... end` syntax. A value of `nothing` can be used to temporarily unset an
244 environment variable (if it is set). When `withenv` returns, the original environment has
245 been restored.
246
247 !!! warning
248 Changing the environment is not thread-safe. For running external commands with a different
249 environment from the parent process, prefer using [`addenv`](@ref) over `withenv`.
250 """
251 function withenv(f, keyvals::Pair{T}...) where T<:AbstractString
252 old = Dict{T,Any}()
253 for (key,val) in keyvals
254 old[key] = get(ENV,key,nothing)
255 val !== nothing ? (ENV[key]=val) : delete!(ENV, key)
256 end
257 try f()
258 finally
259 for (key,val) in old
260 val !== nothing ? (ENV[key]=val) : delete!(ENV, key)
261 end
262 end
263 end
