581 product_high = upper
582 product_middle = product_middle1 + product_middle2
583 if product_middle < product_middle1
584 product_high += 1 # overflow carry
585 end
586 if (product_middle + 1 == 0) && (product_high & 0x1FF == 0x1FF) &&
587 (product_low + v < product_low)
588 return _scale(T, UInt128(v), exp, neg)
589 end
590 upper = product_high
591 lower = product_middle
592 end
593 upperbit = upper >> 63
594 mantissa = upper >> (upperbit + 9)
595 lz += xor(1, upperbit)
596 if (lower == 0) && upper & 0x1FF == 0 && mantissa & 3 == 1
597 return _scale(T, UInt128(v), exp, neg)
598 end
599 mantissa += mantissa & 1
600 mantissa >>= 1
601 if mantissa >= UInt64(1) << 53
602 mantissa = UInt64(1) << 52
603 lz -= 1
604 end
605 mantissa &= ~(UInt64(1) << 52)
606 real_exponent = pow - lz
607 if real_exponent < 1 || real_exponent > 2046
608 return _scale(T, UInt128(v), exp, neg)
609 end
610 mantissa |= real_exponent << 52
611 mantissa |= (UInt64(neg) << 63)
612 return T(Core.bitcast(Float64, mantissa))
613 end

635 function _scale(::Type{T}, v::V, exp, neg) where {T, V <: UInt128}
636 if exp == 23
637 # special-case concluded from https://github.com/JuliaLang/julia/issues/38509
638 x = v * V(1e23)
639 elseif 0 <= exp < 290
640 x = v * exp10(exp)
641 elseif exp < -308 || exp > 308 || v > maxsig(T)
642 # if v is too large, we lose precision by just doing
643 # v / exp10(-exp) since that only promotes to Float64
644 # so detect and re-route to this branch where we widen v
645 y = _widen(v)
646 return _scale(T, y, exp, neg)
647 else
648 x = v / exp10(-exp)
649 end
650 return convert_and_apply_neg(T, x, neg)
651 end

653 const BIGEXP10 = [1 / exp10(BigInt(e)) for e = 309:327]
654 const BIGFLOATS = BigFloat[]
655 const BIGFLOATEXP10 = [exp10(BigFloat(i; precision=256)) for i = 1:308]
656
657 function _scale(::Type{T}, v::V, exp, neg) where {T, V <: BigInt}
658 x = access_threaded(BigFloat, BIGFLOATS)
659
660 ccall((:mpfr_set_z, :libmpfr), Int32,
661 (Ref{BigFloat}, Ref{BigInt}, Int32),
662 x, v, MPFR.ROUNDING_MODE[])
663 if exp < -308
664 # v * (1 / exp10(-exp))
665 if exp < -327
666 y = 1 / exp10(BigInt(-exp))
667 else
668 y = BIGEXP10[-exp - 308]
669 end
670 ccall((:mpfr_mul, :libmpfr), Int32,
671 (Ref{BigFloat}, Ref{BigFloat}, Ref{BigFloat}, Int32),
672 x, x, y, MPFR.ROUNDING_MODE[])
673 elseif exp < 0
674 # v / exp10(-exp)
675 y = BIGFLOATEXP10[-exp]
676 ccall((:mpfr_div, :libmpfr), Int32,
677 (Ref{BigFloat}, Ref{BigFloat}, Ref{BigFloat}, Int32),
678 x, x, y, MPFR.ROUNDING_MODE[])
679 else
680 # v * exp10(V(exp))
681 if exp <= 308
682 y = BIGFLOATEXP10[exp]
683 else
684 y = exp10(BigFloat(exp; precision=256))
685 end
686 ccall((:mpfr_mul, :libmpfr), Int32,
687 (Ref{BigFloat}, Ref{BigFloat}, Ref{BigFloat}, Int32),
688 x, x, y, MPFR.ROUNDING_MODE[])
689 end
690 return convert_and_apply_neg(T, x, neg)
691 end
