116 function find_subranges_sorted(inds, allow_steprange=false)
117 T = allow_steprange ? Union{UnitRange{Int},StepRange{}} : UnitRange{Int}
118 rangelist = T[]
119 outputinds = UnitRange{Int}[]
120 current_step = 0
121 current_base = 1
122 for iind in 1:length(inds)-1
123 next_step = inds[iind+1] - inds[iind]
124 if (next_step == current_step) || (next_step == 0)
125 nothing
126 else
127 if !allow_steprange && next_step != 1
128 #Need to close the range
129 push!(rangelist, inds[current_base]:inds[iind])
130 push!(outputinds, current_base:iind)
131 current_base = iind + 1
132 current_step = 0
133 continue
134 end
135 if current_step === 0
136 # Just set the step (hasnt been set before)
137 current_step = inds[iind+1] - inds[iind]
138 else
139 #Need to close the range
140 if current_step == 1
141 push!(rangelist, inds[current_base]:inds[iind])
142 else
143 push!(rangelist, inds[current_base]:current_step:inds[iind])
144 end
145 push!(outputinds, current_base:iind)
146 current_base = iind + 1
147 current_step = 0
148 continue
149 end
150 end
151 end
152 if current_step == 1 || current_step == 0
153 push!(rangelist, inds[current_base]:last(inds))
154 else
155 push!(rangelist, inds[current_base]:current_step:last(inds))
156 end
157 push!(outputinds, current_base:length(inds))
158
159 return rangelist, outputinds
160 end
