542 code, tlen = dt2.code, dt2.tlen
543 if Parsers.ok(code) && (!ensure_full_buf_consumed || (ensure_full_buf_consumed == ((pos + tlen - 1) == len)))
544 return cb(code, tlen, dt2.val, TIME)
545 end

547 dt3 = Parsers.xparse(DT, buf, pos, len, opts)
548 code, tlen = dt3.code, dt3.tlen
549 if Parsers.ok(code) && (!ensure_full_buf_consumed || (ensure_full_buf_consumed == ((pos + tlen - 1) == len)))
550 return cb(code, tlen, dt3.val, DATETIME)

552 end
553 end
554 bool = Parsers.xparse(Bool, buf, pos, len, opts)
555 code, tlen = bool.code, bool.tlen
556 if Parsers.ok(code) && (!ensure_full_buf_consumed || (ensure_full_buf_consumed == ((pos + tlen - 1) == len)))
557 return cb(code, tlen, bool.val, BOOL)
558 end
559 return cb(code, tlen, nothing, STRING)

610 macro refargs(ex)
611 ex isa Expr || throw(ArgumentError("must pass an expression to @refargs"))
612 (ex.head == :call || ex.head == :function) || throw(ArgumentError("@refargs ex must be function call or definition"))
613 if ex.head == :call
614 for i = 2:length(ex.args)
615 ex.args[i] = Expr(:call, :(CSV.Arg), ex.args[i])
616 end
617 return esc(ex)
618 else # ex.head == :function
619 refs = Expr(:block)
620 fargs = ex.args[1].args
621 for i = 2:length(fargs)
622 arg = fargs[i]
623 (arg isa Symbol || arg.head == :(::)) || throw(ArgumentError("unsupported argument expression: `$arg`"))
624 nm = arg isa Symbol ? arg : arg.args[1]
625 T = arg isa Symbol ? :Any : arg.args[2]
626 push!(refs.args, Expr(:(=), Expr(:(::), nm, T), Expr(:ref, nm)))
627 fargs[i] = Expr(:(::), nm, :(CSV.Arg))
628 end
629 pushfirst!(ex.args[2].args, refs)
630 return esc(ex)
631 end
