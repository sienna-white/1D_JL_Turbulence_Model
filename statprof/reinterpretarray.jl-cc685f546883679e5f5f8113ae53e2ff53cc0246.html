30 reinterpret(T::DataType, A::AbstractArray)

45 1 + 2im

120 """
121 reinterpret(reshape, T, A::AbstractArray{S}) -> B
122
123 Change the type-interpretation of `A` while consuming or adding a "channel dimension."
124
125 If `sizeof(T) = n*sizeof(S)` for `n>1`, `A`'s first dimension must be
126 of size `n` and `B` lacks `A`'s first dimension. Conversely, if `sizeof(S) = n*sizeof(T)` for `n>1`,
127 `B` gets a new first dimension of size `n`. The dimensionality is unchanged if `sizeof(T) == sizeof(S)`.
128
129 !!! compat "Julia 1.6"
130 This method requires at least Julia 1.6.
131
132 # Examples
133
134 ```jldoctest
135 julia> A = [1 2; 3 4]
136 2×2 Matrix{$Int}:
137 1 2
138 3 4
139
140 julia> reinterpret(reshape, Complex{Int}, A) # the result is a vector
141 2-element reinterpret(reshape, Complex{$Int}, ::Matrix{$Int}) with eltype Complex{$Int}:
142 1 + 3im
143 2 + 4im
144
145 julia> a = [(1,2,3), (4,5,6)]
146 2-element Vector{Tuple{$Int, $Int, $Int}}:
147 (1, 2, 3)
148 (4, 5, 6)
149
150 julia> reinterpret(reshape, Int, a) # the result is a matrix
151 3×2 reinterpret(reshape, $Int, ::Vector{Tuple{$Int, $Int, $Int}}) with eltype $Int:
152 1 4
153 2 5
154 3 6
155 ```
156 """

177 function strides(a::ReinterpretArray{T,<:Any,S,<:AbstractArray{S},IsReshaped}) where {T,S,IsReshaped}
178 _checkcontiguous(Bool, a) && return size_to_strides(1, size(a)...)
179 stp = strides(parent(a))
180 els, elp = sizeof(T), sizeof(S)
181 els == elp && return stp # 0dim parent is also handled here.
182 IsReshaped && els < elp && return (1, _checked_strides(stp, els, elp)...)
183 stp[1] == 1 || throw(ArgumentError("Parent must be contiguous in the 1st dimension!"))
184 st′ = _checked_strides(tail(stp), els, elp)
185 return IsReshaped ? st′ : (1, st′...)
186 end

188 @inline function _checked_strides(stp::Tuple, els::Integer, elp::Integer)
189 if elp > els && rem(elp, els) == 0
190 N = div(elp, els)
191 return map(i -> N * i, stp)
192 end
193 drs = map(i -> divrem(elp * i, els), stp)
194 all(i->iszero(i[2]), drs) ||
195 throw(ArgumentError("Parent's strides could not be exactly divided!"))
196 map(first, drs)
197 end

210 function check_writable(a::ReinterpretArray{T, N, S} where N) where {T,S}
211 # `array_subpadding` is relatively expensive (compared to a simple arrayref),
212 # so it is cached in the array. However, it is computable at compile time if,
213 # inference has the types available. By using this form of the check, we can
214 # get the best of both worlds for the success case. If the types were not
215 # available to inference, we simply need to check the field (relatively cheap)
216 # and if they were we should be able to fold this check away entirely.
217 if !a.writable && !array_subpadding(S, T)
218 throw(PaddingError(T, S))
219 end
220 end

390 @inline @propagate_inbounds function _getindex_ra(a::NonReshapedReinterpretArray{T,N,S}, i1::Int, tailinds::TT) where {T,N,S,TT}
391 # Make sure to match the scalar reinterpret if that is applicable
392 if sizeof(T) == sizeof(S) && (fieldcount(T) + fieldcount(S)) == 0
393 if issingletontype(T) # singleton types
394 @boundscheck checkbounds(a, i1, tailinds...)
395 return T.instance
396 end
397 return reinterpret(T, a.parent[i1, tailinds...])
398 else
399 @boundscheck checkbounds(a, i1, tailinds...)
400 ind_start, sidx = divrem((i1-1)*sizeof(T), sizeof(S))
401 # Optimizations that avoid branches
402 if sizeof(T) % sizeof(S) == 0
403 # T is bigger than S and contains an integer number of them
404 n = sizeof(T) ÷ sizeof(S)
405 t = Ref{T}()
406 GC.@preserve t begin
407 sptr = Ptr{S}(unsafe_convert(Ref{T}, t))
408 for i = 1:n
409 s = a.parent[ind_start + i, tailinds...]
410 unsafe_store!(sptr, s, i)
411 end
412 end
413 return t[]
414 elseif sizeof(S) % sizeof(T) == 0
415 # S is bigger than T and contains an integer number of them
416 s = Ref{S}(a.parent[ind_start + 1, tailinds...])
417 GC.@preserve s begin
418 tptr = Ptr{T}(unsafe_convert(Ref{S}, s))
419 return unsafe_load(tptr + sidx)
420 end
421 else
422 i = 1
423 nbytes_copied = 0
424 # This is a bit complicated to deal with partial elements
425 # at both the start and the end. LLVM will fold as appropriate,
426 # once it knows the data layout
427 s = Ref{S}()
428 t = Ref{T}()
429 GC.@preserve s t begin
430 sptr = Ptr{S}(unsafe_convert(Ref{S}, s))
431 tptr = Ptr{T}(unsafe_convert(Ref{T}, t))
432 while nbytes_copied < sizeof(T)
433 s[] = a.parent[ind_start + i, tailinds...]
434 nb = min(sizeof(S) - sidx, sizeof(T)-nbytes_copied)
435 memcpy(tptr + nbytes_copied, sptr + sidx, nb)
436 nbytes_copied += nb
437 sidx = 0
438 i += 1
439 end
440 end
441 return t[]
442 end
443 end
444 end

446 @inline @propagate_inbounds function _getindex_ra(a::ReshapedReinterpretArray{T,N,S}, i1::Int, tailinds::TT) where {T,N,S,TT}
447 # Make sure to match the scalar reinterpret if that is applicable
448 if sizeof(T) == sizeof(S) && (fieldcount(T) + fieldcount(S)) == 0
449 if issingletontype(T) # singleton types
450 @boundscheck checkbounds(a, i1, tailinds...)
451 return T.instance
452 end
453 return reinterpret(T, a.parent[i1, tailinds...])
454 end
455 @boundscheck checkbounds(a, i1, tailinds...)
456 if sizeof(T) >= sizeof(S)
457 t = Ref{T}()
458 GC.@preserve t begin
459 sptr = Ptr{S}(unsafe_convert(Ref{T}, t))
460 if sizeof(T) > sizeof(S)
461 # Extra dimension in the parent array
462 n = sizeof(T) ÷ sizeof(S)
463 if isempty(tailinds) && IndexStyle(a.parent) === IndexLinear()
464 offset = n * (i1 - firstindex(a))
465 for i = 1:n
466 s = a.parent[i + offset]
467 unsafe_store!(sptr, s, i)
468 end
469 else
470 for i = 1:n
471 s = a.parent[i, i1, tailinds...]
472 unsafe_store!(sptr, s, i)
473 end
474 end
475 else
476 # No extra dimension
477 s = a.parent[i1, tailinds...]
478 unsafe_store!(sptr, s)
479 end
480 end
481 return t[]
482 end
483 # S is bigger than T and contains an integer number of them
484 # n = sizeof(S) ÷ sizeof(T)
485 s = Ref{S}()
486 GC.@preserve s begin
487 tptr = Ptr{T}(unsafe_convert(Ref{S}, s))
488 s[] = a.parent[tailinds...]
489 return unsafe_load(tptr, i1)
490 end
491 end

529 @inline @propagate_inbounds function _setindex_ra!(a::NonReshapedReinterpretArray{T,N,S}, v, i1::Int, tailinds::TT) where {T,N,S,TT}
530 v = convert(T, v)::T
531 # Make sure to match the scalar reinterpret if that is applicable
532 if sizeof(T) == sizeof(S) && (fieldcount(T) + fieldcount(S)) == 0
533 if issingletontype(T) # singleton types
534 @boundscheck checkbounds(a, i1, tailinds...)
535 # setindex! is a noop except for the index check
536 else
537 setindex!(a.parent, reinterpret(S, v), i1, tailinds...)
538 end
539 else
540 @boundscheck checkbounds(a, i1, tailinds...)
541 ind_start, sidx = divrem((i1-1)*sizeof(T), sizeof(S))
542 # Optimizations that avoid branches
543 if sizeof(T) % sizeof(S) == 0
544 # T is bigger than S and contains an integer number of them
545 t = Ref{T}(v)
546 GC.@preserve t begin
547 sptr = Ptr{S}(unsafe_convert(Ref{T}, t))
548 n = sizeof(T) ÷ sizeof(S)
549 for i = 1:n
550 s = unsafe_load(sptr, i)
551 a.parent[ind_start + i, tailinds...] = s
552 end
553 end
554 elseif sizeof(S) % sizeof(T) == 0
555 # S is bigger than T and contains an integer number of them
556 s = Ref{S}(a.parent[ind_start + 1, tailinds...])
557 GC.@preserve s begin
558 tptr = Ptr{T}(unsafe_convert(Ref{S}, s))
559 unsafe_store!(tptr + sidx, v)
560 a.parent[ind_start + 1, tailinds...] = s[]
561 end
562 else
563 t = Ref{T}(v)
564 s = Ref{S}()
565 GC.@preserve t s begin
566 tptr = Ptr{UInt8}(unsafe_convert(Ref{T}, t))
567 sptr = Ptr{UInt8}(unsafe_convert(Ref{S}, s))
568 nbytes_copied = 0
569 i = 1
570 # Deal with any partial elements at the start. We'll have to copy in the
571 # element from the original array and overwrite the relevant parts
572 if sidx != 0
573 s[] = a.parent[ind_start + i, tailinds...]
574 nb = min((sizeof(S) - sidx) % UInt, sizeof(T) % UInt)
575 memcpy(sptr + sidx, tptr, nb)
576 nbytes_copied += nb
577 a.parent[ind_start + i, tailinds...] = s[]
578 i += 1
579 sidx = 0
580 end
581 # Deal with the main body of elements
582 while nbytes_copied < sizeof(T) && (sizeof(T) - nbytes_copied) > sizeof(S)
583 nb = min(sizeof(S), sizeof(T) - nbytes_copied)
584 memcpy(sptr, tptr + nbytes_copied, nb)
585 nbytes_copied += nb
586 a.parent[ind_start + i, tailinds...] = s[]
587 i += 1
588 end
589 # Deal with trailing partial elements
590 if nbytes_copied < sizeof(T)
591 s[] = a.parent[ind_start + i, tailinds...]
592 nb = min(sizeof(S), sizeof(T) - nbytes_copied)
593 memcpy(sptr, tptr + nbytes_copied, nb)
594 a.parent[ind_start + i, tailinds...] = s[]
595 end
596 end
597 end
598 end
599 return a
600 end

602 @inline @propagate_inbounds function _setindex_ra!(a::ReshapedReinterpretArray{T,N,S}, v, i1::Int, tailinds::TT) where {T,N,S,TT}
603 v = convert(T, v)::T
604 # Make sure to match the scalar reinterpret if that is applicable
605 if sizeof(T) == sizeof(S) && (fieldcount(T) + fieldcount(S)) == 0
606 if issingletontype(T) # singleton types
607 @boundscheck checkbounds(a, i1, tailinds...)
608 # setindex! is a noop except for the index check
609 else
610 setindex!(a.parent, reinterpret(S, v), i1, tailinds...)
611 end
612 end
613 @boundscheck checkbounds(a, i1, tailinds...)
614 if sizeof(T) >= sizeof(S)
615 t = Ref{T}(v)
616 GC.@preserve t begin
617 sptr = Ptr{S}(unsafe_convert(Ref{T}, t))
618 if sizeof(T) > sizeof(S)
619 # Extra dimension in the parent array
620 n = sizeof(T) ÷ sizeof(S)
621 if isempty(tailinds) && IndexStyle(a.parent) === IndexLinear()
622 offset = n * (i1 - firstindex(a))
623 for i = 1:n
624 s = unsafe_load(sptr, i)
625 a.parent[i + offset] = s
626 end
627 else
628 for i = 1:n
629 s = unsafe_load(sptr, i)
630 a.parent[i, i1, tailinds...] = s
631 end
632 end
633 else # sizeof(T) == sizeof(S)
634 # No extra dimension
635 s = unsafe_load(sptr)
636 a.parent[i1, tailinds...] = s
637 end
638 end
639 else
640 # S is bigger than T and contains an integer number of them
641 s = Ref{S}()
642 GC.@preserve s begin
643 tptr = Ptr{T}(unsafe_convert(Ref{S}, s))
644 s[] = a.parent[tailinds...]
645 unsafe_store!(tptr, v, i1)
646 a.parent[tailinds...] = s[]
647 end
648 end
649 return a
650 end

652 # Padding
653 struct Padding
654 offset::Int # 0-indexed offset of the next valid byte; sizeof(T) indicates trailing padding
655 size::Int # bytes of padding before a valid byte
656 end
657 function intersect(p1::Padding, p2::Padding)
658 start = max(p1.offset, p2.offset)
659 stop = min(p1.offset + p1.size, p2.offset + p2.size)
660 Padding(start, max(0, stop-start))
661 end

672 """
673 CyclePadding(padding, total_size)
674
675 Cylces an iterator of `Padding` structs, restarting the padding at `total_size`.
676 E.g. if `padding` is all the padding in a struct and `total_size` is the total
677 aligned size of that array, `CyclePadding` will correspond to the padding in an
678 infinite vector of such structs.
679 """
680 struct CyclePadding{P}
681 padding::P
682 total_size::Int
683 end
684 eltype(::Type{<:CyclePadding}) = Padding
685 IteratorSize(::Type{<:CyclePadding}) = IsInfinite()
686 isempty(cp::CyclePadding) = isempty(cp.padding)
687 function iterate(cp::CyclePadding)
688 y = iterate(cp.padding)
689 y === nothing && return nothing
690 y[1], (0, y[2])
691 end
692 function iterate(cp::CyclePadding, state::Tuple)
693 y = iterate(cp.padding, tail(state)...)
694 y === nothing && return iterate(cp, (state[1]+cp.total_size,))
695 Padding(y[1].offset+state[1], y[1].size), (state[1], tail(y)...)
696 end
697
698 """
699 Compute the location of padding in an isbits datatype. Recursive over the fields of that type.
700 """
701 @assume_effects :foldable function padding(T::DataType, baseoffset::Int = 0)
702 pads = Padding[]
703 last_end::Int = baseoffset
704 for i = 1:fieldcount(T)
705 offset = baseoffset + Int(fieldoffset(T, i))
706 fT = fieldtype(T, i)
707 append!(pads, padding(fT, offset))
708 if offset != last_end
709 push!(pads, Padding(offset, offset-last_end))
710 end
711 last_end = offset + sizeof(fT)
712 end
713 if 0 < last_end - baseoffset < sizeof(T)
714 push!(pads, Padding(baseoffset + sizeof(T), sizeof(T) - last_end + baseoffset))
715 end
716 return Core.svec(pads...)
717 end
718
719 function CyclePadding(T::DataType)
720 a, s = datatype_alignment(T), sizeof(T)
721 as = s + (a - (s % a)) % a
722 pad = padding(T)
723 if s != as
724 pad = Core.svec(pad..., Padding(s, as - s))
725 end
726 CyclePadding(pad, as)
727 end
728
729 @assume_effects :total function array_subpadding(S, T)
730 lcm_size = lcm(sizeof(S), sizeof(T))
731 s, t = CyclePadding(S), CyclePadding(T)
732 checked_size = 0
733 # use of Stateful harms inference and makes this vulnerable to invalidation
734 (pad, tstate) = let
735 it = iterate(t)
736 it === nothing && return true
737 it
738 end
739 (ps, sstate) = let
740 it = iterate(s)
741 it === nothing && return false
742 it
743 end
744 while checked_size < lcm_size
745 while true
746 # See if there's corresponding padding in S
747 ps.offset > pad.offset && return false
748 intersect(ps, pad) == pad && break
749 ps, sstate = iterate(s, sstate)
750 end
751 checked_size = pad.offset + pad.size
752 pad, tstate = iterate(t, tstate)
753 end
754 return true
755 end

757 @assume_effects :foldable function struct_subpadding(::Type{Out}, ::Type{In}) where {Out, In}
758 padding(Out) == padding(In)
759 end

761 @assume_effects :foldable function packedsize(::Type{T}) where T
762 pads = padding(T)
763 return sizeof(T) - sum((p.size for p ∈ pads), init = 0)
764 end

766 @assume_effects :foldable ispacked(::Type{T}) where T = isempty(padding(T))
767
768 function _copytopacked!(ptr_out::Ptr{Out}, ptr_in::Ptr{In}) where {Out, In}
769 writeoffset = 0
770 for i ∈ 1:fieldcount(In)
771 readoffset = fieldoffset(In, i)
772 fT = fieldtype(In, i)
773 if ispacked(fT)
774 readsize = sizeof(fT)
775 memcpy(ptr_out + writeoffset, ptr_in + readoffset, readsize)
776 writeoffset += readsize
777 else # nested padded type
778 _copytopacked!(ptr_out + writeoffset, Ptr{fT}(ptr_in + readoffset))
779 writeoffset += packedsize(fT)
780 end
781 end
782 end
783
784 function _copyfrompacked!(ptr_out::Ptr{Out}, ptr_in::Ptr{In}) where {Out, In}
785 readoffset = 0
786 for i ∈ 1:fieldcount(Out)
787 writeoffset = fieldoffset(Out, i)
788 fT = fieldtype(Out, i)
789 if ispacked(fT)
790 writesize = sizeof(fT)
791 memcpy(ptr_out + writeoffset, ptr_in + readoffset, writesize)
792 readoffset += writesize
793 else # nested padded type
794 _copyfrompacked!(Ptr{fT}(ptr_out + writeoffset), ptr_in + readoffset)
795 readoffset += packedsize(fT)
796 end
797 end
798 end

800 @inline function _reinterpret(::Type{Out}, x::In) where {Out, In}
801 # handle non-primitive types
802 isbitstype(Out) || throw(ArgumentError("Target type for `reinterpret` must be isbits"))
803 isbitstype(In) || throw(ArgumentError("Source type for `reinterpret` must be isbits"))
804 inpackedsize = packedsize(In)
805 outpackedsize = packedsize(Out)
806 inpackedsize == outpackedsize ||
807 throw(ArgumentError("Packed sizes of types $Out and $In do not match; got $outpackedsize \
808 and $inpackedsize, respectively."))
809 in = Ref{In}(x)
810 out = Ref{Out}()
811 if struct_subpadding(Out, In)
812 # if packed the same, just copy
813 GC.@preserve in out begin
814 ptr_in = unsafe_convert(Ptr{In}, in)
815 ptr_out = unsafe_convert(Ptr{Out}, out)
816 memcpy(ptr_out, ptr_in, sizeof(Out))
817 end
818 return out[]
819 else
820 # mismatched padding
821 GC.@preserve in out begin
822 ptr_in = unsafe_convert(Ptr{In}, in)
823 ptr_out = unsafe_convert(Ptr{Out}, out)
824
825 if fieldcount(In) > 0 && ispacked(Out)
826 _copytopacked!(ptr_out, ptr_in)
827 elseif fieldcount(Out) > 0 && ispacked(In)
828 _copyfrompacked!(ptr_out, ptr_in)
829 else
830 packed = Ref{NTuple{inpackedsize, UInt8}}()
831 GC.@preserve packed begin
832 ptr_packed = unsafe_convert(Ptr{NTuple{inpackedsize, UInt8}}, packed)
833 _copytopacked!(ptr_packed, ptr_in)
834 _copyfrompacked!(ptr_out, ptr_packed)
835 end
836 end
837 end
838 return out[]
839 end
840 end

845 function _mapreduce(f::F, op::OP, style::IndexSCartesian2{K}, A::AbstractArrayOrBroadcasted) where {F,OP,K}
846 inds = eachindex(style, A)
847 n = size(inds)[2]
848 if n == 0
849 return mapreduce_empty_iter(f, op, A, IteratorEltype(A))
850 else
851 return mapreduce_impl(f, op, A, first(inds), last(inds))
852 end

854
855 @noinline function mapreduce_impl(f::F, op::OP, A::AbstractArrayOrBroadcasted,
856 ifirst::SCI, ilast::SCI, blksize::Int) where {F,OP,SCI<:SCartesianIndex2{K}} where K
857 if ilast.j - ifirst.j < blksize
858 # sequential portion
859 @inbounds a1 = A[ifirst]
860 @inbounds a2 = A[SCI(2,ifirst.j)]
861 v = op(f(a1), f(a2))
862 @simd for i = ifirst.i + 2 : K
863 @inbounds ai = A[SCI(i,ifirst.j)]
864 v = op(v, f(ai))
865 end
866 # Remaining columns
867 for j = ifirst.j+1 : ilast.j
868 @simd for i = 1:K
869 @inbounds ai = A[SCI(i,j)]
870 v = op(v, f(ai))
871 end
872 end
873 return v
874 else
875 # pairwise portion
876 jmid = ifirst.j + (ilast.j - ifirst.j) >> 1
877 v1 = mapreduce_impl(f, op, A, ifirst, SCI(K,jmid), blksize)
878 v2 = mapreduce_impl(f, op, A, SCI(1,jmid+1), ilast, blksize)
879 return op(v1, v2)
880 end
881 end

884 mapreduce_impl(f, op, A, ifirst, ilast, pairwise_blocksize(f, op))
