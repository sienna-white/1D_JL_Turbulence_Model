4 An abstract type for which all concrete types expose an interface
5 for working with tabular data.
6
7 An `AbstractDataFrame` is a two-dimensional table with `Symbol`s or strings
8 for column names.
9
10 DataFrames.jl defines two types that are subtypes of `AbstractDataFrame`:
11 [`DataFrame`](@ref) and [`SubDataFrame`](@ref).
12
13 # Indexing and broadcasting
14
15 `AbstractDataFrame` can be indexed by passing two indices specifying
16 row and column selectors. The allowed indices are a superset of indices
17 that can be used for standard arrays. You can also access a single column
18 of an `AbstractDataFrame` using `getproperty` and `setproperty!` functions.
19 Columns can be selected using integers, `Symbol`s, or strings.
20 In broadcasting `AbstractDataFrame` behavior is similar to a `Matrix`.
21
22 A detailed description of `getindex`, `setindex!`, `getproperty`, `setproperty!`,
23 broadcasting and broadcasting assignment for data frames is given in the
24 ["Indexing" section](https://juliadata.github.io/DataFrames.jl/stable/lib/indexing/)
25 of the manual.

603 """

605 describe(df::AbstractDataFrame, stats::Union{Symbol, Pair}...; cols=:)
606
607 Return descriptive statistics for a data frame as a new `DataFrame`
608 where each row represents a variable and each column a summary statistic.
609
610 # Arguments
611 - `df` : the `AbstractDataFrame`
612 - `stats::Union{Symbol, Pair}...` : the summary statistics to report.
613 Arguments can be:
614 - A symbol from the list `:mean`, `:std`, `:min`, `:q25`,
615 `:median`, `:q75`, `:max`, `:sum`, `:eltype`, `:nunique`, `:nuniqueall`, `:first`,
616 `:last`, `:nnonmissing`, and `:nmissing`. The default statistics used are
617 `:mean`, `:min`, `:median`, `:max`, `:nmissing`, and `:eltype`.
618 - `:detailed` as the only `Symbol` argument to return all statistics
619 except `:first`, `:last`, `:sum`, `:nuniqueall`, and `:nnonmissing`.
620 - `:all` as the only `Symbol` argument to return all statistics.
621 - A `function => name` pair where `name` is a `Symbol` or string. This will
622 create a column of summary statistics with the provided name.
623 - `cols` : a keyword argument allowing to select only a subset or transformation
624 of columns from `df` to describe. Can be any column selector or transformation
625 accepted by [`select`](@ref).
626
627 # Details
628 For `Real` columns, compute the mean, standard deviation, minimum, first
629 quantile, median, third quantile, and maximum. If a column does not derive from
630 `Real`, `describe` will attempt to calculate all statistics, using `nothing` as
631 a fall-back in the case of an error.
632
633 When `stats` contains `:nunique`, `describe` will report the
634 number of unique values in a column. If a column's base type derives from `Real`,
635 `:nunique` will return `nothing`s. Use `:nuniqueall` to report the number of
636 unique values in all columns.
637
638 Missing values are filtered in the calculation of all statistics, however the
639 column `:nmissing` will report the number of missing values of that variable
640 and `:nnonmissing` the number of non-missing values.
641
642 If custom functions are provided, they are called repeatedly with the vector
643 corresponding to each column as the only argument. For columns allowing for
644 missing values, the vector is wrapped in a call to `skipmissing`: custom
645 functions must therefore support such objects (and not only vectors), and cannot
646 access missing values.
647
648 Metadata: this function drops all metadata.
649
650 # Examples

704 :first, :last, :eltype]
705
706 if predefined_funs == [:all]
707 predefined_funs = allowed_fields
708 i = findfirst(s -> s == :all, stats)
709 splice!(stats, i, allowed_fields) # insert in the stats vector to get a good order
710 elseif predefined_funs == [:detailed]
711 predefined_funs = [:mean, :std, :min, :q25, :median, :q75,

715 elseif :all in predefined_funs || :detailed in predefined_funs
716 throw(ArgumentError("`:all` and `:detailed` must be the only `Symbol` argument."))
717 elseif !issubset(predefined_funs, allowed_fields)
718 not_allowed = join(setdiff(predefined_funs, allowed_fields), ", :")

720 throw(ArgumentError(":$not_allowed not allowed." * allowed_msg))
721 end
722
723 custom_funs = Any[s[1] => Symbol(s[2]) for s in stats if s isa Pair]
724

726
727 if !allunique(ordered_names)
728 df_ord_names = DataFrame(ordered_names = ordered_names)
729 duplicate_names = unique(ordered_names[nonunique(df_ord_names)])

784 function get_stats(@nospecialize(col::Union{AbstractVector, Base.SkipMissing}),
785 stats::AbstractVector{Symbol})
786 d = Dict{Symbol, Any}()
787
788 if :q25 in stats || :median in stats || :q75 in stats
789 # types that do not support basic arithmetic (like strings) will only fail
790 # after sorting the data, so check this beforehand to fail early
791 T = eltype(col)
792 if isconcretetype(T) && !hasmethod(-, Tuple{T, T})
793 d[:q25] = d[:median] = d[:q75] = nothing
794 else
795 mcol = Base.copymutable(col)
796 if :q25 in stats
797 d[:q25] = try quantile!(mcol, 0.25) catch; nothing; end
798 end
799 if :median in stats
800 d[:median] = try quantile!(mcol, 0.50) catch; nothing; end
801 end
802 if :q75 in stats
803 d[:q75] = try quantile!(mcol, 0.75) catch; nothing; end
804 end
805 end
806 end
807
808 if :min in stats || :max in stats
809 ex = try extrema(col) catch; (nothing, nothing) end
810 d[:min] = ex[1]
811 d[:max] = ex[2]
812 end
813
814 if :mean in stats || :std in stats
815 m = try mean(col) catch end
816 # we can add non-necessary things to d, because we choose what we need
817 # in the main function
818 d[:mean] = m
819
820 if :std in stats
821 d[:std] = try std(col, mean = m) catch end
822 end
823 end
824
825 if :nunique in stats
826 if eltype(col) <: Real
827 d[:nunique] = nothing
828 else
829 d[:nunique] = try length(Set(col)) catch end
830 end
831 end
832
833 if :nuniqueall in stats
834 d[:nuniqueall] = try length(Set(col)) catch end
835 end
836
837 if :sum in stats
838 d[:sum] = try sum(col) catch end
839 end
840
841 return d
842 end

1422 fillcombinations(df::AbstractDataFrame, indexcols;
1423 allowduplicates::Bool=false,
1424 fill=missing)
1425
1426 Generate all combinations of levels of column(s) `indexcols` in data frame `df`.
1427 Levels and their order are determined by the `levels` function
1428 (i.e. unique values sorted lexicographically by default, or a custom set
1429 of levels for e.g. `CategoricalArray` columns), in addition to `missing` if present.

1433
1434 If `allowduplicates=false` (the default) `indexcols` may only contain
1435 unique combinations of `indexcols` values. If `allowduplicates=true`
1436 duplicates are allowed.

1438 $METADATA_FIXED
1439
1440 # Examples
1441
1442 ```jldoctest
1443 julia> df = DataFrame(x=1:2, y='a':'b', z=["x", "y"])

1465 ─────┼──────────────────────
1466 1 │ 1 a x
1467 2 │ 0 b x
1468 3 │ 0 a y
1469 4 │ 2 b y
1470 ```
1471 """
1472 function fillcombinations(df::AbstractDataFrame, indexcols;
1473 allowduplicates::Bool=false, fill=missing)
1474 _check_consistency(df)
1475
1476 colind = index(df)[indexcols]
1477
1478 if length(colind) == 0
1479 throw(ArgumentError("At least one column to fill combinations " *
1480 "must be specified"))
1481 end
1482
1483 # we use hashing algorithm here, because we assume that the tables we work with are not huge
1484 has_duplicates = row_group_slots!(ntuple(i -> df[!, colind[i]], length(colind)),
1485 Val(false), nothing, false, nothing, true)[1] != nrow(df)
1486 if has_duplicates && !allowduplicates
1487 throw(ArgumentError("duplicate combinations of `indexcols` are not " *
1488 "allowed in input when `allowduplicates=false`"))
1489 end
1490
1491 # Create a vector of vectors of unique values in each column
1492 uniquevals = []
1493 for col in colind
1494 # All levels are retained, missing is added only if present
1495 tempcol = levels(df[!, col], skipmissing=false)
1496 push!(uniquevals, tempcol)
1497 end
1498
1499 # make sure we do not overflow in the target data frame size
1500 target_rows = Int(prod(x -> BigInt(length(x)), uniquevals))
1501 if iszero(target_rows)
1502 @assert iszero(nrow(df))
1503 cdf = copy(df)
1504 _drop_all_nonnote_metadata!(cdf)
1505 return cdf
1506 end
1507
1508 # construct expanded columns
1509 out_df = DataFrame()
1510 inner = 1
1511 @assert length(uniquevals) == length(colind)
1512 for (val, cind) in zip(uniquevals, colind)
1513 len = length(val)
1514 target_col = Tables.allocatecolumn(eltype(df[!, cind]), len)
1515 copy!(target_col, val)
1516 last_inner = inner
1517 inner *= len
1518 outer, remv = divrem(target_rows, inner)
1519 @assert iszero(remv)
1520 out_df[!, _names(df)[cind]] = repeat(target_col, inner=last_inner, outer=outer)
1521 end
1522 @assert inner == target_rows
1523
1524 idx_ind = 0
1525 idx_col = ""
1526 while true
1527 idx_col = string("source7249", idx_ind)
1528 columnindex(df, idx_col) == 0 && break
1529 idx_ind += 1
1530 end
1531
1532 if has_duplicates
1533 order_ind = 0
1534 order_col = ""
1535 while true
1536 order_col = string("order9427", order_ind)
1537 columnindex(df, order_col) == 0 && break
1538 order_ind += 1
1539 end
1540 insertcols!(out_df, 1, order_col => 1:nrow(out_df))
1541 out_df = leftjoin(out_df, df; on=_names(df)[colind],
1542 source=idx_col,
1543 matchmissing=:equal)
1544 sort!(out_df, 1)
1545 select!(out_df, 2:ncol(out_df))
1546 else
1547 leftjoin!(out_df, df; on=_names(df)[colind],
1548 source=string("source7249", idx_ind),
1549 matchmissing=:equal)
1550 end
1551
1552 # Replace missing values with the fill
1553 if !ismissing(fill)
1554 mask = out_df[!, end] .== "left_only"
1555 if count(mask) > 0
1556 for n in length(colind)+1:ncol(out_df)-1
1557 tmp_col = out_df[!, n]
1558 out_col = similar(tmp_col,
1559 promote_type(eltype(tmp_col), typeof(fill)),
1560 length(tmp_col))
1561 out_col .= ifelse.(mask, Ref(fill), out_df[!, n])
1562 out_df[!, n] = out_col
1563 end
1564 end
1565 end
1566
1567 # keep only columns from the source in their original order
1568 select!(out_df, _names(df))
1569 _copy_all_note_metadata!(out_df, df)
1570
1571 return out_df
1572 end

1656 cols::Union{Symbol, AbstractVector{Symbol},
1657 AbstractVector{<:AbstractString}}=:setequal,
1658 source::Union{Nothing, Symbol, AbstractString,
1659 Pair{<:Union{Symbol, AbstractString}, <:AbstractVector}}=nothing)

1662

1664

1668 * `:orderequal`: require all data frames to have the same column names and in

2380 # convert a classical permutation to zero terminated cycle
2381 # notation, zeroing the original permutation in the process.
2382 function _compile_permutation!(p::AbstractVector{<:Integer})
2383 firstindex(p) == 1 ||
2384 throw(ArgumentError("Permutation vectors must have 1-based indexing"))
2385 # this length is sufficient because we do not record 1-cycles,
2386 # so the worst case is all 2-cycles. One extra element gives the
2387 # algorithm leeway to defer error detection without unsafe reads.
2388 # trace _compile_permutation!([3,3,1]) for example.
2389 out = similar(p, 3 * length(p) ÷ 2 + 1)
2390 out_len = 0
2391 start = 0
2392 count = length(p)
2393 @inbounds while count > 0
2394 start = findnext(!iszero, p, start + 1)
2395 start isa Int || throw(ArgumentError("Passed vector p is not a valid permutation"))
2396 last_k = p[start]
2397 count -= 1
2398 last_k == start && continue
2399 out_len += 1
2400 out[out_len] = last_k
2401 p[start] = 0
2402 start < last_k <= length(p) || throw(ArgumentError("Passed vector p is not a valid permutation"))
2403 out_len += 1
2404 k = out[out_len] = p[last_k]
2405 while true
2406 count -= 1
2407 p[last_k] = 0
2408 last_k = k
2409 start <= k <= length(p) || throw(ArgumentError("Passed vector p is not a valid permutation"))
2410 out_len += 1
2411 k = out[out_len] = p[k]
2412 k == 0 && break
2413 end
2414 last_k == start || throw(ArgumentError("Passed vector p is not a valid permutation"))
2415 end
2416 return resize!(out, out_len)
2417 end

2419 # Permute a vector `v` based on a permutation `p` listed in zero terminated
2420 # cycle notation. For example, the permutation 1 -> 2, 2 -> 3, 3 -> 1, 4 -> 6,
2421 # 5 -> 5, 6 -> 4 is traditionally expressed as [2, 3, 1, 6, 5, 4] but in cycle
2422 # notation is expressed as [1, 2, 3, 0, 4, 6, 0]
2423 function _cycle_permute!(v::AbstractVector, p::AbstractVector{<:Integer})
2424 i = firstindex(p)
2425 @inbounds while i < lastindex(p)
2426 last_p_i = p[i]
2427 start = v[last_p_i]
2428 while true
2429 i += 1
2430 p_i = p[i]
2431 p_i == 0 && break
2432 v[last_p_i] = v[p_i]
2433 last_p_i = p_i
2434 end
2435 v[last_p_i] = start
2436 i += 1
2437 end
2438 return v
2439 end

2756 function insertcols!(df::AbstractDataFrame, col::ColumnIndex, name_cols::Pair{Symbol}...;
2757 after::Bool=false, makeunique::Bool=false, copycols::Bool=true)
2758 if !is_column_insertion_allowed(df)
2759 throw(ArgumentError("insertcols! is only supported for DataFrame, or for " *
2760 "SubDataFrame created with `:` as column selector"))
2761 end
2762 if !(copycols || df isa DataFrame)
2763 throw(ArgumentError("copycols=false is only allowed if df isa DataFrame "))
2764 end
2765 if col isa SymbolOrString
2766 col_ind = Int(columnindex(df, col))
2767 if col_ind == 0
2768 throw(ArgumentError("column $col does not exist in data frame"))
2769 end
2770 else
2771 col_ind = Int(col)
2772 end
2773
2774 if after
2775 col_ind += 1
2776 end
2777
2778 if !(0 < col_ind <= ncol(df) + 1)
2779 throw(ArgumentError("attempt to insert a column to a data frame with " *
2780 "$(ncol(df)) columns at index $col_ind"))
2781 end
2782
2783 if !makeunique
2784 if !allunique(first.(name_cols))
2785 throw(ArgumentError("Names of columns to be inserted into a data frame " *
2786 "must be unique when `makeunique=true`"))
2787 end
2788 for (n, _) in name_cols
2789 if hasproperty(df, n)
2790 throw(ArgumentError("Column $n is already present in the data frame " *
2791 "which is not allowed when `makeunique=true`"))
2792 end
2793 end
2794 end
2795
2796 if ncol(df) == 0 && df isa DataFrame
2797 target_row_count = -1
2798 else
2799 target_row_count = nrow(df)
2800 end
2801
2802 for (n, v) in name_cols
2803 if v isa AbstractVector
2804 if target_row_count == -1
2805 target_row_count = length(v)
2806 elseif length(v) != target_row_count
2807 if target_row_count == nrow(df)
2808 throw(DimensionMismatch("length of new column $n which is " *
2809 "$(length(v)) must match the number " *
2810 "of rows in data frame ($(nrow(df)))"))
2811 else
2812 throw(DimensionMismatch("all vectors passed to be inserted into " *
2813 "a data frame must have the same length"))
2814 end
2815 end
2816 elseif v isa AbstractArray && ndims(v) > 1
2817 throw(ArgumentError("adding AbstractArray other than AbstractVector as " *
2818 "a column of a data frame is not allowed"))
2819 end
2820 end
2821 if target_row_count == -1
2822 target_row_count = 1
2823 end
2824
2825 start_col_ind = col_ind
2826 for (name, item) in name_cols
2827 if !(item isa AbstractVector)
2828 if item isa Union{AbstractArray{<:Any, 0}, Ref}
2829 x = item[]
2830 item_new = fill!(Tables.allocatecolumn(typeof(x), target_row_count), x)
2831 else
2832 @assert !(item isa AbstractArray)
2833 item_new = fill!(Tables.allocatecolumn(typeof(item), target_row_count), item)
2834 end
2835 elseif item isa AbstractRange
2836 item_new = collect(item)
2837 elseif copycols && df isa DataFrame
2838 item_new = copy(item)
2839 else
2840 item_new = item
2841 end
2842
2843 if df isa DataFrame
2844 dfp = df
2845 else
2846 @assert df isa SubDataFrame
2847 dfp = parent(df)
2848 item_new_orig = item_new
2849 T = eltype(item_new_orig)
2850 item_new = similar(item_new_orig, Union{T, Missing}, nrow(dfp))
2851 fill!(item_new, missing)
2852 item_new[rows(df)] = item_new_orig
2853 end
2854
2855 firstindex(item_new) != 1 && _onebased_check_error()
2856
2857 if ncol(dfp) == 0
2858 dfp[!, name] = item_new
2859 else
2860 if hasproperty(dfp, name)
2861 @assert makeunique
2862 k = 1
2863 while true
2864 nn = Symbol("$(name)_$k")
2865 if !hasproperty(dfp, nn)
2866 name = nn
2867 break
2868 end
2869 k += 1
2870 end
2871 end
2872 insert!(index(dfp), col_ind, name)
2873 insert!(_columns(dfp), col_ind, item_new)
2874 end
2875 col_ind += 1
2876 end
2877
2878 delta = col_ind - start_col_ind
2879 colmetadata_dict = getfield(parent(df), :colmetadata)
2880 if !isnothing(colmetadata_dict) && delta > 0
2881 to_move = Int[i for i in keys(colmetadata_dict) if i >= start_col_ind]
2882 sort!(to_move, rev=true)
2883 for i in to_move
2884 colmetadata_dict[i + delta] = pop!(colmetadata_dict, i)
2885 end
2886 end
2887 _drop_all_nonnote_metadata!(parent(df))
2888 return df
2889 end
