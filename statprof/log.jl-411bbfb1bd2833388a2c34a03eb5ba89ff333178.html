193 @inline function log_proc2(f::Float64,base=Val(:))
194 ## Step 1
195 g = 1.0/(2.0+f)
196 u = 2.0*f*g
197 v = u*u
198
199 ## Step 2
200 q = u*v*@horner(v,
201 0.08333333333333179,
202 0.012500000003771751,
203 0.0022321399879194482,
204 0.0004348877777076146)

206 ## Step 3
207 # based on:
208 # 2(f-u) = 2(f(2+f)-2f)/(2+f) = 2f^2/(2+f) = fu

211 # u2 = (2(f-u1) - f*u1)/(2+f)

214 m_lo = logbL(Float64, base)
215 return fma(m_hi, u, fma(m_lo, u, m_hi*fma(fma(-u,f,2(f-u)), g, q)))
