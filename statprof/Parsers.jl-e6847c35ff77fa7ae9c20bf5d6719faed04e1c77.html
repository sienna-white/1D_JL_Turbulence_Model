96 """
97 `Parsers.Options` is a structure for holding various parsing settings when calling `Parsers.parse`, `Parsers.tryparse`, and `Parsers.xparse`. They include:
98
99 * `sentinel=nothing`: valid values include: `nothing` meaning don't check for sentinel values; `missing` meaning an "empty field" should be considered a sentinel value; or a `Vector{String}` of the various string values that should each be checked as a sentinel value. Note that sentinels will always be checked longest to shortest, with the longest valid match taking precedence.
100 * `openquotechar='"'`: the ascii character that signals a "quoted" field while parsing; subsequent characters will be treated as non-significant until a valid `closequotechar` is detected
101 * `closequotechar='"'`: the ascii character that signals the end of a quoted field
102 * `escapechar='"'`: an ascii character used to "escape" a `closequotechar` within a quoted field
103 * `delim=','`: if `nothing`, no delimiter will be checked for; if a `Char` or `String`, a delimiter will be checked for directly after parsing a value or `closequotechar`; a newline (`\\n`), return (`\\r`), or CRLF (`"\\r\\n"`) are always considered "delimiters", in addition to EOF
104 * `decimal='.'`: an ascii character to be used when parsing float values that separates a decimal value
105 * `trues=nothing`: if `nothing`, `Bool` parsing will only check for the string `true` or an `Integer` value of `1` as valid values for `true`; as a `Vector{String}`, each string value will be checked to indicate a valid `true` value
106 * `falses=nothing`: if `nothing`, `Bool` parsing will only check for the string `false` or an `Integer` value of `0` as valid values for `false`; as a `Vector{String}`, each string value will be checked to indicate a valid `false` value
107 * `dateformat=nothing`: if `nothing`, `Date`, `DateTime`, and `Time` parsing will use a default `Dates.DateFormat` object while parsing; a `String` or `Dates.DateFormat` object can be provided for custom format parsing
108 * `ignorerepeated=false`: if `true`, consecutive delimiter characters or strings will be consumed until a non-delimiter is encountered; if `false`, only a single delimiter character/string will be consumed. Useful for fixed-width delimited files where fields are padded with delimiters
109 * `quoted=true`: whether parsing should check for `openquotechar` and `closequotechar` characters to signal quoted fields
110 * `comment=nothing`: a string which, if matched at the start of a line, will make parsing consume the rest of the line
111 * `ignoreemptylines=false`: after parsing a value, if a newline is detected, another immediately proceeding newline will be checked for and consumed
112 * `stripwhitespace=nothing`: if true, leading and trailing whitespace is stripped from string fields, note that for *quoted* strings however, whitespace is preserved within quotes (but ignored before/after quote characters). To also strip *within* quotes, see `stripquoted`
113 * `stripquoted=false`: if true, whitespace is also stripped within quoted strings. If true, `stripwhitespace` is also set to true.
114 * `groupmark=nothing`: optionally specify a single-byte character denoting the number grouping mark, this allows parsing of numbers that have, e.g., thousand separators (`1,000.00`). When the `groupmark` is ambiguous with the `delim`, the user must quote the number if it contains group marks.
115 * `rounding=RoundNearest`: optionally specify a rounding mode to use when parsing. No rounding means the result will be marked with `INEXACT` code if the value is not exactly representable in the target type.

440 Result(whitespace(false, false, false, true)(typeparser(opts)))(conf, source, pos, len, S)

445 @inline function xparse2(conf::AbstractConf{T}, source::SourceType, pos, len, options=OPTIONS, ::Type{S}=returntype(T)) where {T, S}
446 buf = source isa AbstractString ? codeunits(source) : source
447 if T === Number || supportedtype(T)
448 return _xparse2(conf, buf, pos, len, options, S)
449 else
450 # generic fallback calls Base.tryparse
451 res = _xparse2(DefaultConf{String}(), source, pos, len, options)
452 code = res.code
453 pl = res.val
454 if !Parsers.invalid(code) && !Parsers.sentinel(code)
455 str = getstring(source, pl, options.e)
456 x = Base.tryparse(T, str)
457 if x === nothing
458 return Result{S}(code | INVALID, res.tlen)
459 else
460 return Result{S}(code, res.tlen, x)
461 end
462 else
463 return Result{S}(code, res.tlen)
464 end
465 end
466 end

468 function checkdelim!(source::AbstractVector{UInt8}, pos, len, options::Options)
469 eof(source, pos, len) && return pos
470 delim = options.delim
471 b = peekbyte(source, pos)
472 if !options.flags.ignorerepeated
473 # we're checking for a single appearance of a delimiter
474 match, pos = checktoken(source, pos, len, b, delim)
475 else
476 # keep parsing as long as we keep matching delim
477 while true
478 match, pos = checktoken(source, pos, len, b, delim)
479 (match && !eof(source, pos, len)) || break
480 b = peekbyte(source, pos)
481 end
482 end
483 return pos
484 end

486 @inline function _has_groupmark(opts::Options, code::ReturnCode)
487 if opts.groupmark !== nothing
488 isquoted = (code & QUOTED) != 0
489 if isquoted || (opts.groupmark != opts.delim)
490 return true
491 end
492 end
493 return false
