125 function rehash!(h::OrderedDict{K,V}, newsz::Integer = length(h.slots)) where {K,V}
126 olds = h.slots
127 keys = h.keys
128 vals = h.vals
129 sz = length(olds)
130 newsz = _tablesz(newsz)
131 h.dirty = true
132 count0 = length(h)
133 if count0 == 0
134 resize!(h.slots, newsz)
135 fill!(h.slots, 0)
136 resize!(h.keys, 0)
137 resize!(h.vals, 0)
138 h.ndel = 0
139 return h
140 end
141
142 slots = zeros(Int32, newsz)
143 maxprobe = 0
144
145 if h.ndel > 0
146 ndel0 = h.ndel
147 ptrs = !isbitstype(K) && !isbitsunion(K)
148 to = 1
149 # TODO: to get the best performance we need to avoid reallocating these.
150 # This algorithm actually works in place, unless the dict is modified
151 # due to GC during this process.
152 newkeys = similar(keys, count0)
153 newvals = similar(vals, count0)
154 @inbounds for from = 1:length(keys)
155 if !ptrs || isassigned(keys, from)
156 k = keys[from]
157 hashk = hash(k)%Int
158 isdeleted = false
159 if !ptrs
160 iter = 0
161 index = (hashk & (sz-1)) + 1
162 while iter <= h.maxprobe
163 si = olds[index]
164 si == from && break
165 # if we find si == 0, then the key was deleted and it's slot was reused/overwritten
166 (si == -from || si == 0) && (isdeleted = true; break)
167 index = (index & (sz-1)) + 1
168 iter += 1
169 end
170 iter > h.maxprobe && (isdeleted = true) # Another case where the slot was reused/overwritten
171 end
172 if !isdeleted
173 index0 = index = (hashk & (newsz-1)) + 1
174 while slots[index] != 0
175 index = (index & (newsz-1)) + 1
176 end
177 probe = (index - index0) & (newsz-1)
178 probe > maxprobe && (maxprobe = probe)
179 slots[index] = to
180 newkeys[to] = k
181 newvals[to] = vals[from]
182 to += 1
183 end
184 if h.ndel != ndel0
185 # if items are removed by finalizers, retry
186 return rehash!(h, newsz)
187 end
188 end
189 end
190 h.keys = newkeys
191 h.vals = newvals
192 h.ndel = 0
193 else
194 @inbounds for i = 1:count0
195 k = keys[i]
196 index0 = index = hashindex(k, newsz)
197 while slots[index] != 0
198 index = (index & (newsz-1)) + 1
199 end
200 probe = (index - index0) & (newsz-1)
201 probe > maxprobe && (maxprobe = probe)
202 slots[index] = i
203 if h.ndel > 0

208 end

239 function ht_keyindex(h::OrderedDict{K,V}, key, direct) where {K,V}
240 slots = h.slots
241 sz = length(slots)
242 iter = 0
243 maxprobe = h.maxprobe

245 keys = h.keys
246

265 function ht_keyindex2(h::OrderedDict{K,V}, key) where {K,V}
